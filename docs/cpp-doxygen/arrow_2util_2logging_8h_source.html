144 // To make the logging lib pluggable with other logging libs and make
145 // the implementation unawared by the user, ArrowLog is only a declaration
146 // which hide the implementation into logging.cc file.
147 // In logging.cc, we can choose different log libs using different macros.
148 
149 // This is also a null log which does not output anything.
150 class ARROW_EXPORT ArrowLogBase {
151  public:
152  virtual ~ArrowLogBase() {}
153 
154  virtual bool IsEnabled() const { return false; }
155 
156  template <typename T>
157  ArrowLogBase& operator<<(const T& t) {
158  if (IsEnabled()) {
159  Stream() << t;
160  }
161  return *this;
162  }
163 
164  protected:
165  virtual std::ostream& Stream() = 0;
166 };
