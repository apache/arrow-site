46 // A bit subtle. Numpy has 5 canonical integer types:
47 // (or, rather, type pairs: signed and unsigned)
48 // NPY_BYTE, NPY_SHORT, NPY_INT, NPY_LONG, NPY_LONGLONG
49 // It also has 4 fixed-width integer aliases.
50 // When mapping Arrow integer types to these 4 fixed-width aliases,
51 // we always miss one of the canonical types (even though it may
52 // have the same width as one of the aliases).
53 // Which one depends on the platform...
54 // On a LP64 system, NPY_INT64 maps to NPY_LONG and
55 // NPY_LONGLONG needs to be handled separately.
56 // On a LLP64 system, NPY_INT32 maps to NPY_LONG and
57 // NPY_INT needs to be handled separately.
58 
59 #if NPY_BITSOF_LONG == 32 && NPY_BITSOF_LONGLONG == 64
60 #define NPY_INT64_IS_LONG_LONG 1
61 #else
62 #define NPY_INT64_IS_LONG_LONG 0
63 #endif
64 
65 #if NPY_BITSOF_INT == 32 && NPY_BITSOF_LONG == 64
66 #define NPY_INT32_IS_INT 1
67 #else
68 #define NPY_INT32_IS_INT 0

83 // See above about the missing Numpy integer type numbers
84 inline int fix_numpy_type_num(int type_num) {
85 #if !NPY_INT32_IS_INT && NPY_BITSOF_INT == 32
86  if (type_num == NPY_INT) return NPY_INT32;
87  if (type_num == NPY_UINT) return NPY_UINT32;
88 #endif
89 #if !NPY_INT64_IS_LONG_LONG && NPY_BITSOF_LONGLONG == 64
90  if (type_num == NPY_LONGLONG) return NPY_INT64;
91  if (type_num == NPY_ULONGLONG) return NPY_UINT64;
92 #endif
93  return type_num;
94 }
