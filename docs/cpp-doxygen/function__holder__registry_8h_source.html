36 #define LAMBDA_MAKER(derived) \
37  [](const FunctionNode& node, FunctionHolderPtr* holder) { \
38  std::shared_ptr<derived> derived_instance; \
39  auto status = derived::Make(node, &derived_instance); \
40  if (status.ok()) { \
41  *holder = derived_instance; \
42  } \
43  return status; \
44  }

47 class FunctionHolderRegistry {
48  public:
49  using maker_type = std::function<Status(const FunctionNode&, FunctionHolderPtr*)>;
50  using map_type = std::unordered_map<std::string, maker_type>;
51 
52  static Status Make(const std::string& name, const FunctionNode& node,
53  FunctionHolderPtr* holder) {
54  auto found = makers().find(name);
55  if (found == makers().end()) {
56  return Status::Invalid("function holder not registered for function " + name);
57  }
58 
59  return found->second(node, holder);
60  }
61 
62  private:
63  static map_type& makers() {
64  static map_type maker_map = {
65  {"like", LAMBDA_MAKER(LikeHolder)},
66  {"to_date", LAMBDA_MAKER(ToDateHolder)},
67  {"random", LAMBDA_MAKER(RandomGeneratorHolder)},
68  {"rand", LAMBDA_MAKER(RandomGeneratorHolder)},
69  };
70  return maker_map;
71  }
72 };
