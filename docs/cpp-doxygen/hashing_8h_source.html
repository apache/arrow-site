83 template <typename Scalar, uint64_t AlgNum>
84 struct ScalarHelper<Scalar, AlgNum, enable_if_t<std::is_integral<Scalar>::value>>
85  : public ScalarHelperBase<Scalar, AlgNum> {
86  // ScalarHelper specialization for integers
87 
88  static hash_t ComputeHash(const Scalar& value) {
89  // Faster hash computation for integers.
90 
91  // Two of xxhash's prime multipliers (which are chosen for their
92  // bit dispersion properties)
93  static constexpr uint64_t multipliers[] = {11400714785074694791ULL,
94  14029467366897019727ULL};
95 
96  // Multiplying by the prime number mixes the low bits into the high bits,
97  // then byte-swapping (which is a single CPU instruction) allows the
98  // combined high and low bits to participate in the initial hash table index.
99  auto h = static_cast<hash_t>(value);
100  return BitUtil::ByteSwap(multipliers[AlgNum] * h);
101  }
102 };

129 template <uint64_t AlgNum = 0>
130 hash_t ComputeStringHash(const void* data, int64_t length) {
131  if (ARROW_PREDICT_TRUE(length <= 16)) {
132  // Specialize for small hash strings, as they are quite common as
133  // hash table keys. Even XXH3 isn't quite as fast.
134  auto p = reinterpret_cast<const uint8_t*>(data);
135  auto n = static_cast<uint32_t>(length);
136  if (n <= 8) {
137  if (n <= 3) {
138  if (n == 0) {
139  return 1U;
140  }
141  uint32_t x = (n << 24) ^ (p[0] << 16) ^ (p[n / 2] << 8) ^ p[n - 1];
142  return ScalarHelper<uint32_t, AlgNum>::ComputeHash(x);
143  }
144  // 4 <= length <= 8
145  // We can read the string as two overlapping 32-bit ints, apply
146  // different hash functions to each of them in parallel, then XOR
147  // the results
148  uint32_t x, y;
149  hash_t hx, hy;
150  x = util::SafeLoadAs<uint32_t>(p + n - 4);
151  y = util::SafeLoadAs<uint32_t>(p);
152  hx = ScalarHelper<uint32_t, AlgNum>::ComputeHash(x);
153  hy = ScalarHelper<uint32_t, AlgNum ^ 1>::ComputeHash(y);
154  return n ^ hx ^ hy;
155  }
156  // 8 <= length <= 16
157  // Apply the same principle as above
158  uint64_t x, y;
159  hash_t hx, hy;
160  x = util::SafeLoadAs<uint64_t>(p + n - 8);
161  y = util::SafeLoadAs<uint64_t>(p);
162  hx = ScalarHelper<uint64_t, AlgNum>::ComputeHash(x);
163  hy = ScalarHelper<uint64_t, AlgNum ^ 1>::ComputeHash(y);
164  return n ^ hx ^ hy;
165  }

197 template <typename Payload>
198 class HashTable {
199  public:
200  static constexpr hash_t kSentinel = 0ULL;
201  static constexpr int64_t kLoadFactor = 2UL;
202 
203  struct Entry {
204  hash_t h;
205  Payload payload;
206 
207  // An entry is valid if the hash is different from the sentinel value
208  operator bool() const { return h != kSentinel; }
209  };
210 
211  HashTable(MemoryPool* pool, uint64_t capacity) : entries_builder_(pool) {
212  DCHECK_NE(pool, nullptr);
213  // Minimum of 32 elements
214  capacity = std::max<uint64_t>(capacity, 32UL);
215  capacity_ = BitUtil::NextPower2(capacity);
216  capacity_mask_ = capacity_ - 1;
217  size_ = 0;
218 
219  DCHECK_OK(UpsizeBuffer(capacity_));
220  }
221 
222  // Lookup with non-linear probing
223  // cmp_func should have signature bool(const Payload*).
224  // Return a (Entry*, found) pair.
225  template <typename CmpFunc>
226  std::pair<Entry*, bool> Lookup(hash_t h, CmpFunc&& cmp_func) {
227  auto p = Lookup<DoCompare, CmpFunc>(h, entries_, capacity_mask_,
228  std::forward<CmpFunc>(cmp_func));
229  return {&entries_[p.first], p.second};
230  }
231 
232  template <typename CmpFunc>
233  std::pair<const Entry*, bool> Lookup(hash_t h, CmpFunc&& cmp_func) const {
234  auto p = Lookup<DoCompare, CmpFunc>(h, entries_, capacity_mask_,
235  std::forward<CmpFunc>(cmp_func));
236  return {&entries_[p.first], p.second};
237  }
238 
239  void Insert(Entry* entry, hash_t h, const Payload& payload) {
240  // Ensure entry is empty before inserting
241  assert(!*entry);
242  entry->h = FixHash(h);
243  entry->payload = payload;
244  ++size_;
245 
246  if (ARROW_PREDICT_FALSE(NeedUpsizing())) {
247  // Resize less frequently since it is expensive
248  DCHECK_OK(Upsize(capacity_ * kLoadFactor * 2));
249  }
250  }
251 
252  uint64_t size() const { return size_; }
253 
254  // Visit all non-empty entries in the table
255  // The visit_func should have signature void(const Entry*)
256  template <typename VisitFunc>
257  void VisitEntries(VisitFunc&& visit_func) const {
258  for (uint64_t i = 0; i < capacity_; i++) {
259  const auto& entry = entries_[i];
260  if (entry) {
261  visit_func(&entry);
262  }
263  }
264  }
265 
266  protected:
267  // NoCompare is for when the value is known not to exist in the table
268  enum CompareKind { DoCompare, NoCompare };
269 
270  // The workhorse lookup function
271  template <CompareKind CKind, typename CmpFunc>
272  std::pair<uint64_t, bool> Lookup(hash_t h, const Entry* entries, uint64_t size_mask,
273  CmpFunc&& cmp_func) const {
274  static constexpr uint8_t perturb_shift = 5;
275 
276  uint64_t index, perturb;
277  const Entry* entry;
278 
279  h = FixHash(h);
280  index = h & size_mask;
281  perturb = (h >> perturb_shift) + 1U;
282 
283  while (true) {
284  entry = &entries[index];
285  if (CompareEntry<CKind, CmpFunc>(h, entry, std::forward<CmpFunc>(cmp_func))) {
286  // Found
287  return {index, true};
288  }
289  if (entry->h == kSentinel) {
290  // Empty slot
291  return {index, false};
292  }
293 
294  // Perturbation logic inspired from CPython's set / dict object.
295  // The goal is that all 64 bits of the unmasked hash value eventually
296  // participate in the probing sequence, to minimize clustering.
297  index = (index + perturb) & size_mask;
298  perturb = (perturb >> perturb_shift) + 1U;
299  }
300  }
301 
302  template <CompareKind CKind, typename CmpFunc>
303  bool CompareEntry(hash_t h, const Entry* entry, CmpFunc&& cmp_func) const {
304  if (CKind == NoCompare) {
305  return false;
306  } else {
307  return entry->h == h && cmp_func(&entry->payload);
308  }
309  }
310 
311  bool NeedUpsizing() const {
312  // Keep the load factor <= 1/2
313  return size_ * kLoadFactor >= capacity_;
314  }
315 
316  Status UpsizeBuffer(uint64_t capacity) {
317  RETURN_NOT_OK(entries_builder_.Resize(capacity));
318  entries_ = entries_builder_.mutable_data();
319  memset(static_cast<void*>(entries_), 0, capacity * sizeof(Entry));
320 
321  return Status::OK();
322  }
323 
324  Status Upsize(uint64_t new_capacity) {
325  assert(new_capacity > capacity_);
326  uint64_t new_mask = new_capacity - 1;
327  assert((new_capacity & new_mask) == 0); // it's a power of two
328 
329  // Stash old entries and seal builder, effectively resetting the Buffer
330  const Entry* old_entries = entries_;
331  std::shared_ptr<Buffer> previous;
332  RETURN_NOT_OK(entries_builder_.Finish(&previous));
333  // Allocate new buffer
334  RETURN_NOT_OK(UpsizeBuffer(new_capacity));
335 
336  for (uint64_t i = 0; i < capacity_; i++) {
337  const auto& entry = old_entries[i];
338  if (entry) {
339  // Dummy compare function will not be called
340  auto p = Lookup<NoCompare>(entry.h, entries_, new_mask,
341  [](const Payload*) { return false; });
342  // Lookup<NoCompare> (and CompareEntry<NoCompare>) ensure that an
343  // empty slots is always returned
344  assert(!p.second);
345  entries_[p.first] = entry;
346  }
347  }
348  capacity_ = new_capacity;
349  capacity_mask_ = new_mask;
350 
351  return Status::OK();
352  }
353 
354  hash_t FixHash(hash_t h) const { return (h == kSentinel) ? 42U : h; }
355 
356  // The number of slots available in the hash table array.
357  uint64_t capacity_;
358  uint64_t capacity_mask_;
359  // The number of used slots in the hash table array.
360  uint64_t size_;
361 
362  Entry* entries_;
363  TypedBufferBuilder<Entry> entries_builder_;
364 };

386 template <typename Scalar, template <class> class HashTableTemplateType = HashTable>
387 class ScalarMemoTable : public MemoTable {
388  public:
389  explicit ScalarMemoTable(MemoryPool* pool, int64_t entries = 0)
390  : hash_table_(pool, static_cast<uint64_t>(entries)) {}
391 
392  int32_t Get(const Scalar& value) const {
393  auto cmp_func = [value](const Payload* payload) -> bool {
394  return ScalarHelper<Scalar, 0>::CompareScalars(payload->value, value);
395  };
396  hash_t h = ComputeHash(value);
397  auto p = hash_table_.Lookup(h, cmp_func);
398  if (p.second) {
399  return p.first->payload.memo_index;
400  } else {
401  return kKeyNotFound;
402  }
403  }
404 
405  template <typename Func1, typename Func2>
406  int32_t GetOrInsert(const Scalar& value, Func1&& on_found, Func2&& on_not_found) {
407  auto cmp_func = [value](const Payload* payload) -> bool {
408  return ScalarHelper<Scalar, 0>::CompareScalars(value, payload->value);
409  };
410  hash_t h = ComputeHash(value);
411  auto p = hash_table_.Lookup(h, cmp_func);
412  int32_t memo_index;
413  if (p.second) {
414  memo_index = p.first->payload.memo_index;
415  on_found(memo_index);
416  } else {
417  memo_index = size();
418  hash_table_.Insert(p.first, h, {value, memo_index});
419  on_not_found(memo_index);
420  }
421  return memo_index;
422  }
423 
424  int32_t GetOrInsert(const Scalar& value) {
425  return GetOrInsert(value, [](int32_t i) {}, [](int32_t i) {});
426  }
427 
428  int32_t GetNull() const { return null_index_; }
429 
430  template <typename Func1, typename Func2>
431  int32_t GetOrInsertNull(Func1&& on_found, Func2&& on_not_found) {
432  int32_t memo_index = GetNull();
433  if (memo_index != kKeyNotFound) {
434  on_found(memo_index);
435  } else {
436  null_index_ = memo_index = size();
437  on_not_found(memo_index);
438  }
439  return memo_index;
440  }
441 
442  int32_t GetOrInsertNull() {
443  return GetOrInsertNull([](int32_t i) {}, [](int32_t i) {});
444  }
445 
446  // The number of entries in the memo table +1 if null was added.
447  // (which is also 1 + the largest memo index)
448  int32_t size() const override {
449  return static_cast<int32_t>(hash_table_.size()) + (GetNull() != kKeyNotFound);
450  }
451 
452  // Copy values starting from index `start` into `out_data`
453  void CopyValues(int32_t start, Scalar* out_data) const {
454  hash_table_.VisitEntries([=](const HashTableEntry* entry) {
455  int32_t index = entry->payload.memo_index - start;
456  if (index >= 0) {
457  out_data[index] = entry->payload.value;
458  }
459  });
460  }
461 
462  void CopyValues(Scalar* out_data) const { CopyValues(0, out_data); }
463 
464  protected:
465  struct Payload {
466  Scalar value;
467  int32_t memo_index;
468  };
469

474 
475  hash_t ComputeHash(const Scalar& value) const {

731  void CopyFixedWidthValues(int32_t start, int32_t width_size, int64_t out_size,
732  uint8_t* out_data) const {
733  // This method exists to cope with the fact that the BinaryMemoTable does
734  // not know the fixed width when inserting the null value. The data
735  // buffer hold a zero length string for the null value (if found).
736  //
737  // Thus, the method will properly inject an empty value of the proper width
738  // in the output buffer.
739  //
740  if (start >= size()) {
741  return;
742  }
743 
744  int32_t null_index = GetNull();
745  if (null_index < start) {
746  // Nothing to skip, proceed as usual.
747  CopyValues(start, out_size, out_data);
748  return;
749  }
750 
751  int32_t left_offset = binary_builder_.offset(start);
752 
753  // Ensure that the data length is exactly missing width_size bytes to fit
754  // in the expected output (n_values * width_size).
755 #ifndef NDEBUG
756  int64_t data_length = values_size() - static_cast<size_t>(left_offset);
757  assert(data_length + width_size == out_size);
758  ARROW_UNUSED(data_length);
759 #endif
760 
761  auto in_data = binary_builder_.value_data() + left_offset;
762  // The null use 0-length in the data, slice the data in 2 and skip by
763  // width_size in out_data. [part_1][width_size][part_2]
764  auto null_data_offset = binary_builder_.offset(null_index);
765  auto left_size = null_data_offset - left_offset;
766  if (left_size > 0) {
767  memcpy(out_data, in_data + left_offset, left_size);
768  }
769 
770  auto right_size = values_size() - static_cast<size_t>(null_data_offset);
771  if (right_size > 0) {
772  // skip the null fixed size value.
773  auto out_offset = left_size + width_size;
774  assert(out_data + out_offset + right_size == out_data + out_size);
775  memcpy(out_data + out_offset, in_data + null_data_offset, right_size);
776  }
777  }
