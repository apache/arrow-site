116  Status Append(int8_t next_type) {
117  ARROW_RETURN_NOT_OK(types_builder_.Append(next_type));
118  if (type_id_to_children_[next_type]->length() == kListMaximumElements) {
119  return Status::CapacityError(
120  "a dense UnionArray cannot contain more than 2^31 - 1 elements from a single "
121  "child");
122  }
123  auto offset = static_cast<int32_t>(type_id_to_children_[next_type]->length());
124  ARROW_RETURN_NOT_OK(offsets_builder_.Append(offset));
125  return AppendToBitmap(true);
126  }
127

131  }
