53 template <typename Error>
54 inline bool IsConnectError(const Aws::Client::AWSError<Error>& error) {
55  if (error.GetErrorType() == Aws::Client::CoreErrors::NETWORK_CONNECTION) {
56  return true;
57  }
58  // Sometimes Minio may fail with a 503 error
59  // (exception name: XMinioServerNotInitialized,
60  // message: "Server not initialized, please try again")
61  auto http_code = static_cast<int>(error.GetResponseCode());
62  switch (http_code) {
63  case 502: // Bad gateway
64  case 503: // Service unavailable
65  case 504: // Gateway timeout
66  return true;
67  default:
68  return false;
69  }
70 }

155 class ConnectRetryStrategy : public Aws::Client::RetryStrategy {
156  public:
157  static const int32_t kDefaultRetryInterval = 200; /* milliseconds */
158  static const int32_t kDefaultMaxRetryDuration = 4000; /* milliseconds */
159 
160  explicit ConnectRetryStrategy(int32_t retry_interval = kDefaultRetryInterval,
161  int32_t max_retry_duration = kDefaultMaxRetryDuration)
162  : retry_interval_(retry_interval), max_retry_duration_(max_retry_duration) {}
163 
164  bool ShouldRetry(const Aws::Client::AWSError<Aws::Client::CoreErrors>& error,
165  long attempted_retries) const override { // NOLINT
166  if (!IsConnectError(error)) {
167  // Not a connect error, don't retry
168  return false;
169  }
170  return attempted_retries * retry_interval_ < max_retry_duration_;
171  }

175  long attempted_retries) const override { // NOLINT

177  }

179  protected:

181  int32_t max_retry_duration_;
