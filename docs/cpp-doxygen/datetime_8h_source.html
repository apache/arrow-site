30 // By default, PyDateTimeAPI is a *static* variable. This forces

87 inline int64_t PyDateTime_to_s(PyDateTime_DateTime* pydatetime) {
88  int64_t total_seconds = 0;
89  total_seconds += PyDateTime_DATE_GET_SECOND(pydatetime);
90  total_seconds += PyDateTime_DATE_GET_MINUTE(pydatetime) * 60;
91  total_seconds += PyDateTime_DATE_GET_HOUR(pydatetime) * 3600;
92 
93  return total_seconds +
94  (PyDate_to_ms(reinterpret_cast<PyDateTime_Date*>(pydatetime)) / 1000LL);
95 }
