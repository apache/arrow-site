57 class GANDIVA_EXPORT LiteralNode : public Node {
58  public:
59  LiteralNode(DataTypePtr type, const LiteralHolder& holder, bool is_null)
60  : Node(type), holder_(holder), is_null_(is_null) {}
61 
62  Status Accept(NodeVisitor& visitor) const override { return visitor.Visit(*this); }
63 
64  const LiteralHolder& holder() const { return holder_; }
65 
66  bool is_null() const { return is_null_; }
67 
68  std::string ToString() const override {
69  std::stringstream ss;
70  ss << "(const " << return_type()->ToString() << ") ";
71  if (is_null()) {
72  ss << std::string("null");
73  return ss.str();
74  }
75 
76  ss << gandiva::ToString(holder_);
77  // The default formatter prints in decimal can cause a loss in precision. so,
78  // print in hex. Can't use hexfloat since gcc 4.9 doesn't support it.
79  if (return_type()->id() == arrow::Type::DOUBLE) {
80  double dvalue = arrow::util::get<double>(holder_);
81  uint64_t bits;
82  memcpy(&bits, &dvalue, sizeof(bits));
83  ss << " raw(" << std::hex << bits << ")";
84  } else if (return_type()->id() == arrow::Type::FLOAT) {
85  float fvalue = arrow::util::get<float>(holder_);
86  uint32_t bits;
87  memcpy(&bits, &fvalue, sizeof(bits));
88  ss << " raw(" << std::hex << bits << ")";
89  }
90  return ss.str();
91  }
92 
93  private:
94  LiteralHolder holder_;
95  bool is_null_;
96 };
