237 template <class T>
238 inline void DeserializeThriftUnencryptedMsg(const uint8_t* buf, uint32_t* len,
239  T* deserialized_msg) {
240  // Deserialize msg bytes into c++ thrift msg using memory transport.
241  shared_ptr<ThriftBuffer> tmem_transport(
242  new ThriftBuffer(const_cast<uint8_t*>(buf), *len));
243  apache::thrift::protocol::TCompactProtocolFactoryT<ThriftBuffer> tproto_factory;
244  shared_ptr<apache::thrift::protocol::TProtocol> tproto = //
245  tproto_factory.getProtocol(tmem_transport);
246  try {
247  deserialized_msg->read(tproto.get());
248  } catch (std::exception& e) {
249  std::stringstream ss;
250  ss << "Couldn't deserialize thrift: " << e.what() << "\n";
251  throw ParquetException(ss.str());
252  }
253  uint32_t bytes_left = tmem_transport->available_read();
254  *len = *len - bytes_left;
255 }
256 
257 // Deserialize a thrift message from buf/len. buf/len must at least contain
258 // all the bytes needed to store the thrift message. On return, len will be
259 // set to the actual length of the header.
260 template <class T>
261 inline void DeserializeThriftMsg(const uint8_t* buf, uint32_t* len, T* deserialized_msg,
262  const std::shared_ptr<Decryptor>& decryptor = NULLPTR) {
263  // thrift message is not encrypted
264  if (decryptor == NULLPTR) {
265  DeserializeThriftUnencryptedMsg(buf, len, deserialized_msg);
266  } else { // thrift message is encrypted
267  uint32_t clen;
268  clen = *len;
269  // decrypt
270  std::shared_ptr<ResizableBuffer> decrypted_buffer =
271  std::static_pointer_cast<ResizableBuffer>(AllocateBuffer(
272  decryptor->pool(),
273  static_cast<int64_t>(clen - decryptor->CiphertextSizeDelta())));
274  const uint8_t* cipher_buf = buf;
275  uint32_t decrypted_buffer_len =
276  decryptor->Decrypt(cipher_buf, 0, decrypted_buffer->mutable_data());
277  if (decrypted_buffer_len <= 0) {
278  throw ParquetException("Couldn't decrypt buffer\n");
279  }
280  *len = decrypted_buffer_len + decryptor->CiphertextSizeDelta();
281  DeserializeThriftMsg(decrypted_buffer->data(), &decrypted_buffer_len,
282  deserialized_msg);
283  }

313  template <class T>
314  int64_t Serialize(const T* obj, ArrowOutputStream* out,
315  const std::shared_ptr<Encryptor>& encryptor = NULLPTR) {
316  uint8_t* out_buffer;
317  uint32_t out_length;
318  SerializeToBuffer(obj, &out_length, &out_buffer);
319 
320  // obj is not encrypted
321  if (encryptor == NULLPTR) {
322  PARQUET_THROW_NOT_OK(out->Write(out_buffer, out_length));
323  return static_cast<int64_t>(out_length);
324  } else { // obj is encrypted
325  return SerializeEncryptedObj(out, out_buffer, out_length, encryptor);
326  }
327  }
328 
329  private:
330  template <class T>
331  void SerializeObject(const T* obj) {
332  try {
333  mem_buffer_->resetBuffer();
334  obj->write(protocol_.get());
335  } catch (std::exception& e) {
336  std::stringstream ss;
337  ss << "Couldn't serialize thrift: " << e.what() << "\n";
338  throw ParquetException(ss.str());
339  }
340  }
341 
342  int64_t SerializeEncryptedObj(ArrowOutputStream* out, uint8_t* out_buffer,
343  uint32_t out_length,
344  const std::shared_ptr<Encryptor>& encryptor) {
345  std::shared_ptr<ResizableBuffer> cipher_buffer =
346  std::static_pointer_cast<ResizableBuffer>(AllocateBuffer(
347  encryptor->pool(),
348  static_cast<int64_t>(encryptor->CiphertextSizeDelta() + out_length)));
349  int cipher_buffer_len =
350  encryptor->Encrypt(out_buffer, out_length, cipher_buffer->mutable_data());
351 
352  PARQUET_THROW_NOT_OK(out->Write(cipher_buffer->data(), cipher_buffer_len));
353  return static_cast<int64_t>(cipher_buffer_len);
354  }
355 
356  shared_ptr<ThriftBuffer> mem_buffer_;
357  shared_ptr<apache::thrift::protocol::TProtocol> protocol_;
358 };
