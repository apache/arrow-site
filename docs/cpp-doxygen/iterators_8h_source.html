30 // Visit the Python sequence, calling the given callable on each element. If
31 // the callable returns a non-OK status, iteration stops and the status is
32 // returned.
33 //
34 // The call signature for Visitor must be
35 //
36 // Visit(PyObject* obj, int64_t index, bool* keep_going)
37 //
38 // If keep_going is set to false, the iteration terminates
39 template <class VisitorFunc>
40 inline Status VisitSequenceGeneric(PyObject* obj, VisitorFunc&& func) {
41  // VisitorFunc may set to false to terminate iteration
42  bool keep_going = true;
43 
44  if (PyArray_Check(obj)) {
45  PyArrayObject* arr_obj = reinterpret_cast<PyArrayObject*>(obj);
46  if (PyArray_NDIM(arr_obj) != 1) {
47  return Status::Invalid("Only 1D arrays accepted");
48  }
49 
50  if (PyArray_DESCR(arr_obj)->type_num == NPY_OBJECT) {
51  // It's an array object, we can fetch object pointers directly
52  const Ndarray1DIndexer<PyObject*> objects(arr_obj);
53  for (int64_t i = 0; keep_going && i < objects.size(); ++i) {
54  RETURN_NOT_OK(func(objects[i], i, &keep_going));
55  }
56  return Status::OK();
57  }
58  // It's a non-object array, fall back on regular sequence access.
59  // (note PyArray_GETITEM() is slightly different: it returns standard
60  // Python types, not Numpy scalar types)
61  // This code path is inefficient: callers should implement dedicated
62  // logic for non-object arrays.
63  }
64  if (PySequence_Check(obj)) {
65  if (PyList_Check(obj) || PyTuple_Check(obj)) {
66  // Use fast item access
67  const Py_ssize_t size = PySequence_Fast_GET_SIZE(obj);
68  for (Py_ssize_t i = 0; keep_going && i < size; ++i) {
69  PyObject* value = PySequence_Fast_GET_ITEM(obj, i);
70  RETURN_NOT_OK(func(value, static_cast<int64_t>(i), &keep_going));
71  }
72  } else {
73  // Regular sequence: avoid making a potentially large copy
74  const Py_ssize_t size = PySequence_Size(obj);
75  RETURN_IF_PYERROR();
76  for (Py_ssize_t i = 0; keep_going && i < size; ++i) {
77  OwnedRef value_ref(PySequence_ITEM(obj, i));
78  RETURN_IF_PYERROR();
79  RETURN_NOT_OK(func(value_ref.obj(), static_cast<int64_t>(i), &keep_going));
80  }
81  }
82  } else {
83  return Status::TypeError("Object is not a sequence");
84  }
85  return Status::OK();
86 }

88 // Visit sequence with no null mask
89 template <class VisitorFunc>
90 inline Status VisitSequence(PyObject* obj, VisitorFunc&& func) {
91  return VisitSequenceGeneric(
92  obj, [&func](PyObject* value, int64_t i /* unused */, bool* keep_going) {
93  return func(value, keep_going);
94  });
95 }

98 template <class VisitorFunc>
99 inline Status VisitSequenceMasked(PyObject* obj, PyObject* mo, VisitorFunc&& func) {
100  if (mo == nullptr || !PyArray_Check(mo)) {
101  return Status::Invalid("Null mask must be NumPy array");
102  }
103 
104  PyArrayObject* mask = reinterpret_cast<PyArrayObject*>(mo);
105  if (PyArray_NDIM(mask) != 1) {
106  return Status::Invalid("Mask must be 1D array");
107  }
108 
109  const Py_ssize_t obj_size = PySequence_Size(obj);
110  if (PyArray_SIZE(mask) != static_cast<int64_t>(obj_size)) {
111  return Status::Invalid("Mask was a different length from sequence being converted");
112  }
113 
114  const int dtype = fix_numpy_type_num(PyArray_DESCR(mask)->type_num);
115  if (dtype == NPY_BOOL) {
116  Ndarray1DIndexer<uint8_t> mask_values(mask);
117 
118  return VisitSequenceGeneric(
119  obj, [&func, &mask_values](PyObject* value, int64_t i, bool* keep_going) {
120  return func(value, mask_values[i], keep_going);
121  });
122  } else {
123  return Status::Invalid("Mask must be boolean dtype");
124  }
125 }

127 // Like IterateSequence, but accepts any generic iterable (including
128 // non-restartable iterators, e.g. generators).
129 //
130 // The call signature for VisitorFunc must be Visit(PyObject*, bool*
131 // keep_going). If keep_going is set to false, the iteration terminates
132 template <class VisitorFunc>
133 inline Status VisitIterable(PyObject* obj, VisitorFunc&& func) {
134  if (PySequence_Check(obj)) {
135  // Numpy arrays fall here as well
136  return VisitSequence(obj, std::forward<VisitorFunc>(func));
137  }
138  // Fall back on the iterator protocol
139  OwnedRef iter_ref(PyObject_GetIter(obj));
140  PyObject* iter = iter_ref.obj();
141  RETURN_IF_PYERROR();
142  PyObject* value;
143 
144  bool keep_going = true;
145  while (keep_going && (value = PyIter_Next(iter))) {
146  OwnedRef value_ref(value);
147  RETURN_NOT_OK(func(value_ref.obj(), &keep_going));
148  }
149  RETURN_IF_PYERROR(); // __next__() might have raised
150  return Status::OK();
151 }
