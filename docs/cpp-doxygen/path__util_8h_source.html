90 // Join the components of an abstract path.
91 template <class StringIt>
92 std::string JoinAbstractPath(StringIt it, StringIt end) {
93  std::string path;
94  for (; it != end; ++it) {
95  if (!path.empty()) {
96  path += kSep;
97  }
98  path += *it;
99  }
100  return path;
101 }

103 template <class StringRange>
104 std::string JoinAbstractPath(const StringRange& range) {
