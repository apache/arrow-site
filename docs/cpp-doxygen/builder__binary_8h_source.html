129  Status AppendValues(const std::vector<std::string>& values,
130  const uint8_t* valid_bytes = NULLPTR) {
131  std::size_t total_length = std::accumulate(
132  values.begin(), values.end(), 0ULL,
133  [](uint64_t sum, const std::string& str) { return sum + str.size(); });
134  ARROW_RETURN_NOT_OK(Reserve(values.size()));
135  ARROW_RETURN_NOT_OK(value_data_builder_.Reserve(total_length));

167  Status AppendValues(const char** values, int64_t length,
168  const uint8_t* valid_bytes = NULLPTR) {
169  std::size_t total_length = 0;
170  std::vector<std::size_t> value_lengths(length);
171  bool have_null_value = false;
172  for (int64_t i = 0; i < length; ++i) {
173  if (values[i] != NULLPTR) {
174  auto value_length = strlen(values[i]);
175  value_lengths[i] = value_length;
176  total_length += value_length;
177  } else {
178  have_null_value = true;
179  }
180  }
181  ARROW_RETURN_NOT_OK(Reserve(length));
182  ARROW_RETURN_NOT_OK(ReserveData(total_length));
183 
184  if (valid_bytes) {
185  int64_t valid_bytes_offset = 0;
186  for (int64_t i = 0; i < length; ++i) {
187  UnsafeAppendNextOffset();
188  if (valid_bytes[i]) {
189  if (values[i]) {
190  value_data_builder_.UnsafeAppend(reinterpret_cast<const uint8_t*>(values[i]),
191  value_lengths[i]);
192  } else {
193  UnsafeAppendToBitmap(valid_bytes + valid_bytes_offset,
194  i - valid_bytes_offset);
195  UnsafeAppendToBitmap(false);
196  valid_bytes_offset = i + 1;
197  }
198  }
199  }
200  UnsafeAppendToBitmap(valid_bytes + valid_bytes_offset, length - valid_bytes_offset);
201  } else {
202  if (have_null_value) {
203  std::vector<uint8_t> valid_vector(length, 0);
204  for (int64_t i = 0; i < length; ++i) {
205  UnsafeAppendNextOffset();
206  if (values[i]) {
207  value_data_builder_.UnsafeAppend(reinterpret_cast<const uint8_t*>(values[i]),
208  value_lengths[i]);
209  valid_vector[i] = 1;
210  }
211  }
212  UnsafeAppendToBitmap(valid_vector.data(), length);
213  } else {
214  for (int64_t i = 0; i < length; ++i) {
215  UnsafeAppendNextOffset();
216  value_data_builder_.UnsafeAppend(reinterpret_cast<const uint8_t*>(values[i]),
217  value_lengths[i]);
218  }
219  UnsafeAppendToBitmap(NULLPTR, length);
220  }
221  }
222  return Status::OK();
223  }

257  // Write final offset (values length)

264  ARROW_RETURN_NOT_OK(null_bitmap_builder_.Finish(&null_bitmap));
265

515 class ARROW_EXPORT ChunkedBinaryBuilder {
516  public:
517  explicit ChunkedBinaryBuilder(int32_t max_chunk_value_length,
518  MemoryPool* pool ARROW_MEMORY_POOL_DEFAULT);
519 
520  ChunkedBinaryBuilder(int32_t max_chunk_value_length, int32_t max_chunk_length,
521  MemoryPool* pool ARROW_MEMORY_POOL_DEFAULT);
522 
523  virtual ~ChunkedBinaryBuilder() = default;
524 
525  Status Append(const uint8_t* value, int32_t length) {
526  if (ARROW_PREDICT_FALSE(length + builder_->value_data_length() >
527  max_chunk_value_length_)) {
528  if (builder_->value_data_length() == 0) {
529  // The current item is larger than max_chunk_size_;
530  // this chunk will be oversize and hold *only* this item
531  ARROW_RETURN_NOT_OK(builder_->Append(value, length));
532  return NextChunk();
533  }
534  // The current item would cause builder_->value_data_length() to exceed
535  // max_chunk_size_, so finish this chunk and append the current item to the next
536  // chunk
537  ARROW_RETURN_NOT_OK(NextChunk());
538  return Append(value, length);
539  }
540 
541  if (ARROW_PREDICT_FALSE(builder_->length() == max_chunk_length_)) {
542  // The current item would cause builder_->length() to exceed max_chunk_length_, so
543  // finish this chunk and append the current item to the next chunk
544  ARROW_RETURN_NOT_OK(NextChunk());
545  }
546 
547  return builder_->Append(value, length);
548  }
549 
550  Status Append(const util::string_view& value) {
551  return Append(reinterpret_cast<const uint8_t*>(value.data()),
552  static_cast<int32_t>(value.size()));
553  }
554 
555  Status AppendNull() {
556  if (ARROW_PREDICT_FALSE(builder_->length() == max_chunk_length_)) {
557  ARROW_RETURN_NOT_OK(NextChunk());
558  }
559  return builder_->AppendNull();
560  }
561 
562  Status Reserve(int64_t values);
563 
564  virtual Status Finish(ArrayVector* out);
565 
566  protected:
567  Status NextChunk();
568 
569  // maximum total character data size per chunk
570  int64_t max_chunk_value_length_;
571 
572  // maximum elements allowed per chunk
573  int64_t max_chunk_length_ = kListMaximumElements;
574 
575  // when Reserve() would cause builder_ to exceed its max_chunk_length_,
576  // add to extra_capacity_ instead and wait to reserve until the next chunk
577  int64_t extra_capacity_ = 0;
578 
579  std::unique_ptr<BinaryBuilder> builder_;
580  std::vector<std::shared_ptr<Array>> chunks_;
581 };
