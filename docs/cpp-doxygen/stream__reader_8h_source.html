242  template <typename ReaderType, typename ReadType = typename ReaderType::T, typename T>
243  void ReadOptional(optional<T>* v) {
244  const auto& node = nodes_[column_index_];
245  auto reader = static_cast<ReaderType*>(column_readers_[column_index_++].get());
246  int16_t def_level;
247  int16_t rep_level;
248  ReadType tmp;
249  int64_t values_read;
250 
251  reader->ReadBatch(kBatchSizeOne, &def_level, &rep_level, &tmp, &values_read);
252 
253  if (values_read == 1) {
254  *v = T(tmp);
255  } else if ((values_read == 0) && (def_level == 0)) {
256  v->reset();
257  } else {
258  ThrowReadFailedException(node);
259  }
260  }
