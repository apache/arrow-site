54 class PARQUET_EXPORT ApplicationVersion {
55  public:
56  // Known Versions with Issues
57  static const ApplicationVersion& PARQUET_251_FIXED_VERSION();
58  static const ApplicationVersion& PARQUET_816_FIXED_VERSION();
59  static const ApplicationVersion& PARQUET_CPP_FIXED_STATS_VERSION();
60  static const ApplicationVersion& PARQUET_MR_FIXED_STATS_VERSION();
61  // Regular expression for the version format
62  // major . minor . patch unknown - prerelease.x + build info
63  // Eg: 1.5.0ab-cdh5.5.0+cd
64  static constexpr char const* VERSION_FORMAT =
65  "^(\\d+)\\.(\\d+)\\.(\\d+)([^-+]*)?(?:-([^+]*))?(?:\\+(.*))?$";
66  // Regular expression for the application format
67  // application_name version VERSION_FORMAT (build build_name)
68  // Eg: parquet-cpp version 1.5.0ab-xyz5.5.0+cd (build abcd)
69  static constexpr char const* APPLICATION_FORMAT =
70  "(.*?)\\s*(?:(version\\s*(?:([^(]*?)\\s*(?:\\(\\s*build\\s*([^)]*?)\\s*\\))?)?)?)";
71 
72  // Application that wrote the file. e.g. "IMPALA"
73  std::string application_;
74  // Build name
75  std::string build_;
76 
77  // Version of the application that wrote the file, expressed as
78  // (<major>.<minor>.<patch>). Unmatched parts default to 0.
79  // "1.2.3" => {1, 2, 3}
80  // "1.2" => {0, 0, 0}
81  // "1.2-cdh5" => {0, 0, 0}
82  // TODO (majetideepak): Implement support for pre_release
83  struct {
84  int major;
85  int minor;
86  int patch;
87  std::string unknown;
88  std::string pre_release;
89  std::string build_info;
90  } version;
91 
92  ApplicationVersion() {}
93  explicit ApplicationVersion(const std::string& created_by);
94  ApplicationVersion(const std::string& application, int major, int minor, int patch);
95 
96  // Returns true if version is strictly less than other_version
97  bool VersionLt(const ApplicationVersion& other_version) const;
98 
99  // Returns true if version is strictly less than other_version
100  bool VersionEq(const ApplicationVersion& other_version) const;
101 
102  // Checks if the Version has the correct statistics for a given column
103  bool HasCorrectStatistics(Type::type primitive, EncodedStatistics& statistics,
104  SortOrder::type sort_order = SortOrder::SIGNED) const;
105 };
