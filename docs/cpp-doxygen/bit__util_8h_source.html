132 // Returns the smallest power of two that contains v. If v is already a
133 // power of two, it is returned as is.
134 static inline int64_t NextPower2(int64_t n) {
135  // Taken from
136  // http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
137  n--;
138  n |= n >> 1;
139  n |= n >> 2;
140  n |= n >> 4;
141  n |= n >> 8;
142  n |= n >> 16;
143  n |= n >> 32;
144  n++;
145  return n;
146 }

198 // Returns the 'num_bits' least-significant bits of 'v'.
199 static inline uint64_t TrailingBits(uint64_t v, int num_bits) {
200  if (ARROW_PREDICT_FALSE(num_bits == 0)) return 0;
201  if (ARROW_PREDICT_FALSE(num_bits >= 64)) return v;
202  int n = 64 - num_bits;
203  return (v << n) >> n;

207 static inline int CountLeadingZeros(uint32_t value) {
208 #if defined(__clang__) || defined(__GNUC__)
209  if (value == 0) return 32;
210  return static_cast<int>(__builtin_clz(value));
211 #elif defined(_MSC_VER)
212  unsigned long index; // NOLINT
213  if (_BitScanReverse(&index, static_cast<unsigned long>(value))) { // NOLINT
214  return 31 - static_cast<int>(index);
215  } else {
216  return 32;
217  }
218 #else
219  int bitpos = 0;
220  while (value != 0) {
221  value >>= 1;
222  ++bitpos;
223  }
224  return 32 - bitpos;
225 #endif
226 }

329 // Write the swapped bytes into dst. Src and dst cannot overlap.
330 static inline void ByteSwap(void* dst, const void* src, int len) {
331  switch (len) {
332  case 1:
333  *reinterpret_cast<int8_t*>(dst) = *reinterpret_cast<const int8_t*>(src);
334  return;
335  case 2:
336  *reinterpret_cast<int16_t*>(dst) = ByteSwap(*reinterpret_cast<const int16_t*>(src));
337  return;
338  case 4:
339  *reinterpret_cast<int32_t*>(dst) = ByteSwap(*reinterpret_cast<const int32_t*>(src));
340  return;
341  case 8:
342  *reinterpret_cast<int64_t*>(dst) = ByteSwap(*reinterpret_cast<const int64_t*>(src));
343  return;
344  default:
345  break;
346  }
347 
348  auto d = reinterpret_cast<uint8_t*>(dst);
349  auto s = reinterpret_cast<const uint8_t*>(src);
350  for (int i = 0; i < len; ++i) {
351  d[i] = s[len - i - 1];
352  }
353 }

440 static inline void SetBitTo(uint8_t* bits, int64_t i, bool bit_is_set) {
441  // https://graphics.stanford.edu/~seander/bithacks.html
442  // "Conditionally set or clear bits without branching"
443  // NOTE: this seems to confuse Valgrind as it reads from potentially
444  // uninitialized memory
445  bits[i / 8] ^= static_cast<uint8_t>(-static_cast<uint8_t>(bit_is_set) ^ bits[i / 8]) &
446  kBitmask[i % 8];
447 }

450 static inline void SetBitsTo(uint8_t* bits, int64_t start_offset, int64_t length,
451  bool bits_are_set) {
452  if (length == 0) return;
453 
454  const auto i_begin = start_offset;
455  const auto i_end = start_offset + length;
456  const uint8_t fill_byte = static_cast<uint8_t>(-static_cast<uint8_t>(bits_are_set));
457 
458  const auto bytes_begin = i_begin / 8;
459  const auto bytes_end = i_end / 8 + 1;
460 
461  const auto first_byte_mask = kPrecedingBitmask[i_begin % 8];
462  const auto last_byte_mask = kTrailingBitmask[i_end % 8];
463 
464  if (bytes_end == bytes_begin + 1) {
465  // set bits within a single byte
466  const auto only_byte_mask =
467  i_end % 8 == 0 ? first_byte_mask
468  : static_cast<uint8_t>(first_byte_mask | last_byte_mask);
469  bits[bytes_begin] &= only_byte_mask;
470  bits[bytes_begin] |= static_cast<uint8_t>(fill_byte & ~only_byte_mask);
471  return;
472  }
473 
474  // set/clear trailing bits of first byte
475  bits[bytes_begin] &= first_byte_mask;
476  bits[bytes_begin] |= static_cast<uint8_t>(fill_byte & ~first_byte_mask);
477 
478  if (bytes_end - bytes_begin > 2) {
479  // set/clear whole bytes
480  std::memset(bits + bytes_begin + 1, fill_byte,
481  static_cast<size_t>(bytes_end - bytes_begin - 2));
482  }
483 
484  if (i_end % 8 == 0) return;
485 
486  // set/clear leading bits of last byte
487  bits[bytes_end - 1] &= last_byte_mask;
488  bits[bytes_end - 1] |= static_cast<uint8_t>(fill_byte & ~last_byte_mask);
489 }

538 class BitmapWriter {
539  // A sequential bitwise writer that preserves surrounding bit values.
540 
541  public:
542  BitmapWriter(uint8_t* bitmap, int64_t start_offset, int64_t length)
543  : bitmap_(bitmap), position_(0), length_(length) {
544  byte_offset_ = start_offset / 8;
545  bit_mask_ = BitUtil::kBitmask[start_offset % 8];
546  if (length > 0) {
547  current_byte_ = bitmap[byte_offset_];
548  } else {
549  current_byte_ = 0;
550  }
551  }
552 
553  void Set() { current_byte_ |= bit_mask_; }
554 
555  void Clear() { current_byte_ &= bit_mask_ ^ 0xFF; }
556 
557  void Next() {
558  bit_mask_ = static_cast<uint8_t>(bit_mask_ << 1);
559  ++position_;
560  if (bit_mask_ == 0) {
561  // Finished this byte, need advancing
562  bit_mask_ = 0x01;
563  bitmap_[byte_offset_++] = current_byte_;
564  if (ARROW_PREDICT_TRUE(position_ < length_)) {
565  current_byte_ = bitmap_[byte_offset_];
566  }
567  }
568  }
569 
570  void Finish() {
571  // Store current byte if we didn't went past bitmap storage
572  if (length_ > 0 && (bit_mask_ != 0x01 || position_ < length_)) {
573  bitmap_[byte_offset_] = current_byte_;
574  }
575  }
576 
577  int64_t position() const { return position_; }
578 
579  private:
580  uint8_t* bitmap_;
581  int64_t position_;
582  int64_t length_;
583 
584  uint8_t current_byte_;
585  uint8_t bit_mask_;
586  int64_t byte_offset_;
587 };

613  ++position_;

616  bit_mask_ = 0x01;
617  bitmap_[byte_offset_++] = current_byte_;
618  current_byte_ = 0;
619  }

625  bitmap_[byte_offset_] = current_byte_;
626  }
627  }

645 template <class Generator>
646 void GenerateBits(uint8_t* bitmap, int64_t start_offset, int64_t length, Generator&& g) {
647  if (length == 0) {
648  return;
649  }
650  uint8_t* cur = bitmap + start_offset / 8;
651  uint8_t bit_mask = BitUtil::kBitmask[start_offset % 8];
652  uint8_t current_byte = *cur & BitUtil::kPrecedingBitmask[start_offset % 8];
653 
654  for (int64_t index = 0; index < length; ++index) {
655  const bool bit = g();
656  current_byte = bit ? (current_byte | bit_mask) : current_byte;
657  bit_mask = static_cast<uint8_t>(bit_mask << 1);
658  if (bit_mask == 0) {
659  bit_mask = 1;
660  *cur++ = current_byte;
661  current_byte = 0;
662  }
663  }
664  if (bit_mask != 1) {
665  *cur++ = current_byte;
666  }
667 }

669 // Like GenerateBits(), but unrolls its main loop for higher performance.
670 
671 template <class Generator>
672 void GenerateBitsUnrolled(uint8_t* bitmap, int64_t start_offset, int64_t length,
673  Generator&& g) {
674  if (length == 0) {
675  return;
676  }
677  uint8_t current_byte;
678  uint8_t* cur = bitmap + start_offset / 8;
679  const uint64_t start_bit_offset = start_offset % 8;
680  uint8_t bit_mask = BitUtil::kBitmask[start_bit_offset];
681  int64_t remaining = length;
682 
683  if (bit_mask != 0x01) {
684  current_byte = *cur & BitUtil::kPrecedingBitmask[start_bit_offset];
685  while (bit_mask != 0 && remaining > 0) {
686  current_byte = g() ? (current_byte | bit_mask) : current_byte;
687  bit_mask = static_cast<uint8_t>(bit_mask << 1);
688  --remaining;
689  }
690  *cur++ = current_byte;
691  }
692 
693  int64_t remaining_bytes = remaining / 8;
694  while (remaining_bytes-- > 0) {
695  current_byte = 0;
696  current_byte = g() ? current_byte | 0x01 : current_byte;
697  current_byte = g() ? current_byte | 0x02 : current_byte;
698  current_byte = g() ? current_byte | 0x04 : current_byte;
699  current_byte = g() ? current_byte | 0x08 : current_byte;
700  current_byte = g() ? current_byte | 0x10 : current_byte;
701  current_byte = g() ? current_byte | 0x20 : current_byte;
702  current_byte = g() ? current_byte | 0x40 : current_byte;
703  current_byte = g() ? current_byte | 0x80 : current_byte;
704  *cur++ = current_byte;
705  }
706 
707  int64_t remaining_bits = remaining % 8;
708  if (remaining_bits) {
709  current_byte = 0;
710  bit_mask = 0x01;
711  while (remaining_bits-- > 0) {
712  current_byte = g() ? (current_byte | bit_mask) : current_byte;
713  bit_mask = static_cast<uint8_t>(bit_mask << 1);
714  }
715  *cur++ = current_byte;
716  }
717 }

719 // A function that visits each bit in a bitmap and calls a visitor function with a
720 // boolean representation of that bit. This is intended to be analogous to
721 // GenerateBits.
722 template <class Visitor>
723 void VisitBits(const uint8_t* bitmap, int64_t start_offset, int64_t length,
724  Visitor&& visit) {

732 // Like VisitBits(), but unrolls its main loop for better performance.
733 template <class Visitor>
734 void VisitBitsUnrolled(const uint8_t* bitmap, int64_t start_offset, int64_t length,
735  Visitor&& visit) {
736  if (length == 0) {
737  return;
738  }
739 
740  // Start by visiting any bits preceding the first full byte.
741  int64_t num_bits_before_full_bytes =
742  BitUtil::RoundUpToMultipleOf8(start_offset) - start_offset;
743  // Truncate num_bits_before_full_bytes if it is greater than length.
744  if (num_bits_before_full_bytes > length) {
745  num_bits_before_full_bytes = length;
746  }
747  // Use the non loop-unrolled VisitBits since we don't want to add branches
748  VisitBits<Visitor>(bitmap, start_offset, num_bits_before_full_bytes, visit);
749 
750  // Shift the start pointer to the first full byte and compute the
751  // number of full bytes to be read.
752  const uint8_t* first_full_byte = bitmap + BitUtil::CeilDiv(start_offset, 8);
753  const int64_t num_full_bytes = (length - num_bits_before_full_bytes) / 8;
754 
755  // Iterate over each full byte of the input bitmap and call the visitor in
756  // a loop-unrolled manner.
757  for (int64_t byte_index = 0; byte_index < num_full_bytes; ++byte_index) {
758  // Get the current bit-packed byte value from the bitmap.
759  const uint8_t byte = *(first_full_byte + byte_index);
760 
761  // Execute the visitor function on each bit of the current byte.
762  visit(BitUtil::GetBitFromByte(byte, 0));
763  visit(BitUtil::GetBitFromByte(byte, 1));
764  visit(BitUtil::GetBitFromByte(byte, 2));
765  visit(BitUtil::GetBitFromByte(byte, 3));
766  visit(BitUtil::GetBitFromByte(byte, 4));
767  visit(BitUtil::GetBitFromByte(byte, 5));
768  visit(BitUtil::GetBitFromByte(byte, 6));
769  visit(BitUtil::GetBitFromByte(byte, 7));
770  }
771 
772  // Write any leftover bits in the last byte.
773  const int64_t num_bits_after_full_bytes = (length - num_bits_before_full_bytes) % 8;
774  VisitBits<Visitor>(first_full_byte + num_full_bytes, 0, num_bits_after_full_bytes,
775  visit);
776 }

910  typename Word =
911  typename internal::call_traits::argument_type<0, Visitor&&>::value_type>
912  static int64_t VisitWords(const Bitmap (&bitmaps_arg)[N], Visitor&& visitor) {
913  constexpr int64_t kBitWidth = sizeof(Word) * 8;
914 
915  // local, mutable variables which will be sliced/decremented to represent consumption:
916  Bitmap bitmaps[N];
917  int64_t offsets[N];
918  int64_t bit_length = BitLength(bitmaps_arg, N);
919  View<Word> words[N];
920  for (size_t i = 0; i < N; ++i) {
921  bitmaps[i] = bitmaps_arg[i];
922  offsets[i] = bitmaps[i].template word_offset<Word>();
923  assert(offsets[i] >= 0 && offsets[i] < kBitWidth);
924  words[i] = bitmaps[i].template words<Word>();
925  }
926 
927  auto consume = [&](int64_t consumed_bits) {
928  for (size_t i = 0; i < N; ++i) {
929  bitmaps[i] = bitmaps[i].Slice(consumed_bits, bit_length - consumed_bits);
930  offsets[i] = bitmaps[i].template word_offset<Word>();
931  assert(offsets[i] >= 0 && offsets[i] < kBitWidth);
932  words[i] = bitmaps[i].template words<Word>();
933  }
934  bit_length -= consumed_bits;
935  };
936 
937  std::array<Word, N> visited_words;
938  visited_words.fill(0);
939 
940  if (bit_length <= kBitWidth * 2) {
941  // bitmaps fit into one or two words so don't bother with optimization
942  while (bit_length > 0) {
943  auto leading_bits = std::min(bit_length, kBitWidth);
944  SafeLoadWords(bitmaps, 0, leading_bits, false, &visited_words);
945  visitor(visited_words);
946  consume(leading_bits);
947  }
948  return 0;
949  }

952  int64_t min_offset = *std::min_element(offsets, offsets + N);
953  if (max_offset > 0) {
954  // consume leading bits
955  auto leading_bits = kBitWidth - min_offset;
956  SafeLoadWords(bitmaps, 0, leading_bits, true, &visited_words);
957  visitor(visited_words);
958  consume(leading_bits);
959  }
960  assert(*std::min_element(offsets, offsets + N) == 0);
961 
962  int64_t whole_word_count = bit_length / kBitWidth;
963  assert(whole_word_count >= 1);
964 
965  if (min_offset == max_offset) {
966  // all offsets were identical, all leading bits have been consumed
967  assert(
968  std::all_of(offsets, offsets + N, [](int64_t offset) { return offset == 0; }));
969 
970  for (int64_t word_i = 0; word_i < whole_word_count; ++word_i) {
971  for (size_t i = 0; i < N; ++i) {
972  visited_words[i] = words[i][word_i];
973  }
974  visitor(visited_words);
975  }
976  consume(whole_word_count * kBitWidth);
977  } else {
978  // leading bits from potentially incomplete words have been consumed
979 
980  // word_i such that words[i][word_i] and words[i][word_i + 1] are lie entirely
981  // within the bitmap for all i
982  for (int64_t word_i = 0; word_i < whole_word_count - 1; ++word_i) {
983  for (size_t i = 0; i < N; ++i) {
984  if (offsets[i] == 0) {
985  visited_words[i] = words[i][word_i];
986  } else {

988  visited_words[i] |= words[i][word_i + 1] << (kBitWidth - offsets[i]);
989  }
990  }

992  }
993  consume((whole_word_count - 1) * kBitWidth);

998  consume(kBitWidth);
999  }

1002  if (bit_length > 0) {
1003  SafeLoadWords(bitmaps, 0, bit_length, false, &visited_words);
1004  visitor(visited_words);
1005  }
