42 // GCC can be told that a certain branch is not likely to be taken (for
43 // instance, a CHECK failure), and use that information in static analysis.
44 // Giving it this information can help it optimize for the common case in
45 // the absence of better information (ie. -fprofile-arcs).
46 //
47 #if defined(__GNUC__)
48 #define ARROW_PREDICT_FALSE(x) (__builtin_expect(x, 0))
49 #define ARROW_PREDICT_TRUE(x) (__builtin_expect(!!(x), 1))
50 #define ARROW_NORETURN __attribute__((noreturn))
51 #define ARROW_PREFETCH(addr) __builtin_prefetch(addr)
52 #elif defined(_MSC_VER)
53 #define ARROW_NORETURN __declspec(noreturn)
54 #define ARROW_PREDICT_FALSE(x) x
55 #define ARROW_PREDICT_TRUE(x) x
56 #define ARROW_PREFETCH(addr)
57 #else
58 #define ARROW_NORETURN
59 #define ARROW_PREDICT_FALSE(x) x
60 #define ARROW_PREDICT_TRUE(x) x

106 // macros to disable padding
107 // these macros are portable across different compilers and platforms
108 //[https://github.com/google/flatbuffers/blob/master/include/flatbuffers/flatbuffers.h#L1355]
109 #if !defined(MANUALLY_ALIGNED_STRUCT)
110 #if defined(_MSC_VER)
111 #define MANUALLY_ALIGNED_STRUCT(alignment) \
112  __pragma(pack(1)); \
113  struct __declspec(align(alignment))
114 #define STRUCT_END(name, size) \
115  __pragma(pack()); \
116  static_assert(sizeof(name) == size, "compiler breaks packing rules")
117 #elif defined(__GNUC__) || defined(__clang__)
118 #define MANUALLY_ALIGNED_STRUCT(alignment) \
119  _Pragma("pack(1)") struct __attribute__((aligned(alignment)))
120 #define STRUCT_END(name, size) \
121  _Pragma("pack()") static_assert(sizeof(name) == size, "compiler breaks packing rules")
122 #else
123 #error Unknown compiler, please define structure alignment macros
124 #endif
