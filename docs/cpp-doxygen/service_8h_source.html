68 // Manages a connection to a HiveServer2 server. Primarily used to create
69 // new sessions via OpenSession.
70 //
71 // Service objects are created using Service::Connect(). They must
72 // have Close called on them before they can be deleted.
73 //
74 // This class is not thread-safe.
75 //
76 // Example:
77 // unique_ptr<Service> service;
78 // if (Service::Connect(host, port, protocol_version, &service).ok()) {
79 // // do some work
80 // service->Close();
81 // }
82 class ARROW_EXPORT Service {
83  public:
84  // Creates a new connection to a HS2 service at the given host and port. If
85  // conn_timeout > 0, connection attempts will timeout after conn_timeout ms, otherwise
86  // no timeout is used. protocol_version is the HiveServer2 protocol to use, and
87  // determines whether the results returned by operations from this service are row or
88  // column oriented. Only column oriented protocols are currently supported.
89  //
90  // The client calling Connect has ownership of the new Service that is created.
91  // Executing RPCs with a Session or Operation corresponding to a particular
92  // Service after that Service has been closed or deleted in undefined.
93  static Status Connect(const std::string& host, int port, int conn_timeout,
94  ProtocolVersion protocol_version,
95  std::unique_ptr<Service>* service);
96 
97  ~Service();
98 
99  // Closes the connection. Must be called before the service is deleted. May be
100  // safely called on an invalid or already closed service - will only return an
101  // error if the service is open but the close rpc fails.
102  Status Close();
103 
104  // Returns true iff this service has an active connection to the HiveServer2 server.
105  bool IsConnected() const;
106 
107  // Set the send and receive timeout for Thrift RPCs in ms. 0 indicates no timeout,
108  // negative values are ignored.
109  void SetRecvTimeout(int timeout);
110  void SetSendTimeout(int timeout);
111 
112  // Opens a new HS2 session using this service.
113  // The client calling OpenSession has ownership of the Session that is created.
114  // Operations on the Session are undefined once it is closed.
115  Status OpenSession(const std::string& user, const HS2ClientConfig& config,
116  std::unique_ptr<Session>* session) const;
117 
118  private:
119  ARROW_DISALLOW_COPY_AND_ASSIGN(Service);
120 
121  // Hides Thrift objects from the header.
122  struct ServiceImpl;
123 
124  Service(const std::string& host, int port, int conn_timeout,
125  ProtocolVersion protocol_version);
126 
127  // Opens the connection to the server. Called by Connect before new service is returned
128  // to the user. Must be called before OpenSession.
129  Status Open();
130 
131  std::string host_;
132  int port_;
133  int conn_timeout_;
134

136  std::shared_ptr<ThriftRPC> rpc_;
137 };
