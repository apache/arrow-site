198  template <typename ValuesIter, typename ValidIter>
199  enable_if_t<!std::is_pointer<ValidIter>::value, Status> AppendValues(
200  ValuesIter values_begin, ValuesIter values_end, ValidIter valid_begin) {
201  static_assert(!internal::is_null_pointer<ValidIter>::value,
202  "Don't pass a NULLPTR directly as valid_begin, use the 2-argument "
203  "version instead");
204  int64_t length = static_cast<int64_t>(std::distance(values_begin, values_end));
205  ARROW_RETURN_NOT_OK(Reserve(length));
206  data_builder_.UnsafeAppend(values_begin, values_end);
207  null_bitmap_builder_.UnsafeAppend<true>(
208  length, [&valid_begin]() -> bool { return *valid_begin++; });
209  length_ = null_bitmap_builder_.length();
210  null_count_ = null_bitmap_builder_.false_count();
211  return Status::OK();
212  }
