31 class ProjectorCacheKey {
32  public:
33  ProjectorCacheKey(SchemaPtr schema, std::shared_ptr<Configuration> configuration,
34  ExpressionVector expression_vector, SelectionVector::Mode mode)
35  : schema_(schema), configuration_(configuration), mode_(mode), uniqifier_(0) {
36  static const int kSeedValue = 4;
37  size_t result = kSeedValue;
38  for (auto& expr : expression_vector) {
39  std::string expr_as_string = expr->ToString();
40  expressions_as_strings_.push_back(expr_as_string);
41  boost::hash_combine(result, expr_as_string);
42  UpdateUniqifier(expr_as_string);
43  }
44  boost::hash_combine(result, mode);
45  boost::hash_combine(result, configuration->Hash());
46  boost::hash_combine(result, schema_->ToString());
47  boost::hash_combine(result, uniqifier_);
48  hash_code_ = result;
49  }
50

103  void UpdateUniqifier(const std::string& expr) {
104  if (uniqifier_ == 0) {
105  // caching of expressions with re2 patterns causes lock contention. So, use
106  // multiple instances to reduce contention.
107  if (expr.find(" like(") != std::string::npos) {
108  uniqifier_ = std::hash<std::thread::id>()(std::this_thread::get_id()) % 16;
109  }
110  }
111  }
