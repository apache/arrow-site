167 static inline Result<std::shared_ptr<Buffer>> WriteFlatbufferBuilder(
168  flatbuffers::FlatBufferBuilder& fbb) {
169  int32_t size = fbb.GetSize();
170 
171  std::shared_ptr<Buffer> result;
172  RETURN_NOT_OK(AllocateBuffer(default_memory_pool(), size, &result));
173 
174  uint8_t* dst = result->mutable_data();
175  memcpy(dst, fbb.GetBufferPointer(), size);
176  return result;
177 }
