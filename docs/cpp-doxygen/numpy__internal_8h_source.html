37 template <typename T>
38 class Ndarray1DIndexer {
39  public:
40  typedef int64_t size_type;
41 
42  Ndarray1DIndexer() : arr_(NULLPTR), data_(NULLPTR) {}
43 
44  explicit Ndarray1DIndexer(PyArrayObject* arr) : Ndarray1DIndexer() {
45  arr_ = arr;
46  DCHECK_EQ(1, PyArray_NDIM(arr)) << "Only works with 1-dimensional arrays";
47  Py_INCREF(arr);
48  data_ = reinterpret_cast<uint8_t*>(PyArray_DATA(arr));
49  stride_ = PyArray_STRIDES(arr)[0];
50  }
51 
52  ~Ndarray1DIndexer() { Py_XDECREF(arr_); }
53 
54  int64_t size() const { return PyArray_SIZE(arr_); }
55 
56  T* data() const { return data_; }
57 
58  bool is_strided() const { return stride_ != sizeof(T); }
59 
60  T& operator[](size_type index) {
61  return *reinterpret_cast<T*>(data_ + index * stride_);
62  }
63  const T& operator[](size_type index) const {
64  return *reinterpret_cast<const T*>(data_ + index * stride_);
65  }
66 
67  private:
68  PyArrayObject* arr_;
69  uint8_t* data_;
70  int64_t stride_;
71 };

76 static inline std::string GetNumPyTypeName(int npy_type) {
77 #define TYPE_CASE(TYPE, NAME) \
78  case NPY_##TYPE: \
79  return NAME;
80 
81  switch (npy_type) {
82  TYPE_CASE(BOOL, "bool")
83  TYPE_CASE(INT8, "int8")
84  TYPE_CASE(INT16, "int16")
85  TYPE_CASE(INT32, "int32")
86  TYPE_CASE(INT64, "int64")
87 #if !NPY_INT32_IS_INT
88  TYPE_CASE(INT, "intc")
89 #endif
90 #if !NPY_INT64_IS_LONG_LONG
91  TYPE_CASE(LONGLONG, "longlong")
92 #endif
93  TYPE_CASE(UINT8, "uint8")
94  TYPE_CASE(UINT16, "uint16")
95  TYPE_CASE(UINT32, "uint32")
96  TYPE_CASE(UINT64, "uint64")
97 #if !NPY_INT32_IS_INT
98  TYPE_CASE(UINT, "uintc")
99 #endif
100 #if !NPY_INT64_IS_LONG_LONG
101  TYPE_CASE(ULONGLONG, "ulonglong")
102 #endif
103  TYPE_CASE(FLOAT16, "float16")
104  TYPE_CASE(FLOAT32, "float32")
105  TYPE_CASE(FLOAT64, "float64")
106  TYPE_CASE(DATETIME, "datetime64")
107  TYPE_CASE(TIMEDELTA, "timedelta64")
108  TYPE_CASE(OBJECT, "object")
109  TYPE_CASE(VOID, "void")
110  default:
111  break;
112  }
113 
114 #undef TYPE_CASE
115  std::stringstream ss;
116  ss << "unrecognized type (" << npy_type << ") in GetNumPyTypeName";
117  return ss.str();
118 }
