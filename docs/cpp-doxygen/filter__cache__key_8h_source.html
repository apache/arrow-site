30 class FilterCacheKey {

32  FilterCacheKey(SchemaPtr schema, std::shared_ptr<Configuration> configuration,
33  Expression& expression)
34  : schema_(schema), configuration_(configuration), uniqifier_(0) {
35  static const int kSeedValue = 4;
36  size_t result = kSeedValue;
37  expression_as_string_ = expression.ToString();
38  UpdateUniqifier(expression_as_string_);
39  boost::hash_combine(result, expression_as_string_);
40  boost::hash_combine(result, configuration);
41  boost::hash_combine(result, schema_->ToString());
42  boost::hash_combine(result, uniqifier_);
43  hash_code_ = result;
44  }
