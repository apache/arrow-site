53 #define HANDLE_SIGPIPE(s, fd_) \
54  do { \
55  Status _s = (s); \
56  if (!_s.ok()) { \
57  if (errno == EPIPE || errno == EBADF || errno == ECONNRESET) { \
58  ARROW_LOG(WARNING) \
59  << "Received SIGPIPE, BAD FILE DESCRIPTOR, or ECONNRESET when " \
60  "sending a message to client on fd " \
61  << fd_ \
62  << ". " \
63  "The client on the other end may have hung up."; \
64  } else { \
65  return _s; \
66  } \
67  } \
68  } while (0);
