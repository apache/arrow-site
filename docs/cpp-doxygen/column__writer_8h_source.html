211 template <int64_t UnitPerDay, int64_t NanosecondsPerUnit>
212 inline void ArrowTimestampToImpalaTimestamp(const int64_t time, Int96* impala_timestamp) {
213  int64_t julian_days = (time / UnitPerDay) + kJulianEpochOffsetDays;
214  (*impala_timestamp).value[2] = (uint32_t)julian_days;
215 
216  int64_t last_day_units = time % UnitPerDay;
217  auto last_day_nanos = last_day_units * NanosecondsPerUnit;
218  // impala_timestamp will be unaligned every other entry so do memcpy instead
219  // of assign and reinterpret cast to avoid undefined behavior.
220  std::memcpy(impala_timestamp, &last_day_nanos, sizeof(int64_t));
221 }
