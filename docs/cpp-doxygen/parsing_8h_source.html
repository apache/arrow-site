58  bool operator()(const char* s, size_t length, value_type* out) {
59  if (length == 1) {
60  // "0" or "1"?
61  if (s[0] == '0') {
62  *out = false;
63  return true;
64  }
65  if (s[0] == '1') {
66  *out = true;
67  return true;
68  }
69  return false;
70  }
71  if (length == 4) {
72  // "true"?
73  *out = true;
74  return ((s[0] == 't' || s[0] == 'T') && (s[1] == 'r' || s[1] == 'R') &&
75  (s[2] == 'u' || s[2] == 'U') && (s[3] == 'e' || s[3] == 'E'));
76  }
77  if (length == 5) {
78  // "false"?
79  *out = false;
80  return ((s[0] == 'f' || s[0] == 'F') && (s[1] == 'a' || s[1] == 'A') &&
81  (s[2] == 'l' || s[2] == 'L') && (s[3] == 's' || s[3] == 'S') &&
82  (s[4] == 'e' || s[4] == 'E'));
83  }
84  return false;
85  }

132 inline uint8_t ParseDecimalDigit(char c) { return static_cast<uint8_t>(c - '0'); }

134 #define PARSE_UNSIGNED_ITERATION(C_TYPE) \
135  if (length > 0) { \
136  uint8_t digit = ParseDecimalDigit(*s++); \
137  result = static_cast<C_TYPE>(result * 10U); \
138  length--; \
139  if (ARROW_PREDICT_FALSE(digit > 9U)) { \
140  /* Non-digit */ \
141  return false; \
142  } \
143  result = static_cast<C_TYPE>(result + digit); \
144  }
145 
146 #define PARSE_UNSIGNED_ITERATION_LAST(C_TYPE) \
147  if (length > 0) { \
148  if (ARROW_PREDICT_FALSE(result > std::numeric_limits<C_TYPE>::max() / 10U)) { \
149  /* Overflow */ \
150  return false; \
151  } \
152  uint8_t digit = ParseDecimalDigit(*s++); \
153  result = static_cast<C_TYPE>(result * 10U); \
154  C_TYPE new_result = static_cast<C_TYPE>(result + digit); \
155  if (ARROW_PREDICT_FALSE(--length > 0)) { \
156  /* Too many digits */ \
157  return false; \
158  } \
159  if (ARROW_PREDICT_FALSE(digit > 9U)) { \
160  /* Non-digit */ \
161  return false; \
162  } \
163  if (ARROW_PREDICT_FALSE(new_result < result)) { \
164  /* Overflow */ \
165  return false; \
166  } \
167  result = new_result; \
168  }

288 template <class ARROW_TYPE>
289 class StringToSignedIntConverterMixin {
290  public:
291  using value_type = typename ARROW_TYPE::c_type;
292  using unsigned_type = typename std::make_unsigned<value_type>::type;
293 
294  explicit StringToSignedIntConverterMixin(const std::shared_ptr<DataType>& = NULLPTR) {}
295 
296  bool operator()(const char* s, size_t length, value_type* out) {
297  static constexpr unsigned_type max_positive =
298  static_cast<unsigned_type>(std::numeric_limits<value_type>::max());
299  // Assuming two's complement
300  static constexpr unsigned_type max_negative = max_positive + 1;
301  bool negative = false;
302  unsigned_type unsigned_value = 0;
303 
304  if (ARROW_PREDICT_FALSE(length == 0)) {
305  return false;
306  }
307  if (*s == '-') {
308  negative = true;
309  s++;
310  if (--length == 0) {
311  return false;
312  }
313  }
314  // Skip leading zeros
315  while (length > 0 && *s == '0') {
316  length--;
317  s++;
318  }
319  if (!ARROW_PREDICT_TRUE(detail::ParseUnsigned(s, length, &unsigned_value))) {
320  return false;
321  }
322  if (negative) {
323  if (ARROW_PREDICT_FALSE(unsigned_value > max_negative)) {
324  return false;
325  }
326  // To avoid both compiler warnings (with unsigned negation)
327  // and undefined behaviour (with signed negation overflow),
328  // use the expanded formula for 2's complement negation.
329  *out = static_cast<value_type>(~unsigned_value + 1);
330  } else {
331  if (ARROW_PREDICT_FALSE(unsigned_value > max_positive)) {
332  return false;
333  }
334  *out = static_cast<value_type>(unsigned_value);
335  }
336  return true;
337  }
338 };

360 template <>
361 class StringConverter<TimestampType> {
362  public:
363  using value_type = TimestampType::c_type;
364 
365  explicit StringConverter(const std::shared_ptr<DataType>& type)
366  : unit_(checked_cast<TimestampType*>(type.get())->unit()) {}
367 
368  bool operator()(const char* s, size_t length, value_type* out) {
369  // We allow the following formats:
370  // - "YYYY-MM-DD"
371  // - "YYYY-MM-DD[ T]hh"
372  // - "YYYY-MM-DD[ T]hhZ"
373  // - "YYYY-MM-DD[ T]hh:mm"
374  // - "YYYY-MM-DD[ T]hh:mmZ"
375  // - "YYYY-MM-DD[ T]hh:mm:ss"
376  // - "YYYY-MM-DD[ T]hh:mm:ssZ"
377  // UTC is always assumed, and the DataType's timezone is ignored.
378  arrow_vendored::date::year_month_day ymd;
379  if (ARROW_PREDICT_FALSE(length < 10)) {
380  return false;
381  }
382  if (length == 10) {
383  if (ARROW_PREDICT_FALSE(!ParseYYYY_MM_DD(s, &ymd))) {
384  return false;
385  }
386  return ConvertTimePoint(arrow_vendored::date::sys_days(ymd), out);
387  }
388  if (ARROW_PREDICT_FALSE(s[10] != ' ') && ARROW_PREDICT_FALSE(s[10] != 'T')) {
389  return false;
390  }
391  if (s[length - 1] == 'Z') {
392  --length;
393  }
394  if (length == 13) {
395  if (ARROW_PREDICT_FALSE(!ParseYYYY_MM_DD(s, &ymd))) {
396  return false;
397  }
398  std::chrono::duration<value_type> seconds;
399  if (ARROW_PREDICT_FALSE(!ParseHH(s + 11, &seconds))) {
400  return false;
401  }
402  return ConvertTimePoint(arrow_vendored::date::sys_days(ymd) + seconds, out);
403  }
404  if (length == 16) {
405  if (ARROW_PREDICT_FALSE(!ParseYYYY_MM_DD(s, &ymd))) {
406  return false;
407  }
408  std::chrono::duration<value_type> seconds;
409  if (ARROW_PREDICT_FALSE(!ParseHH_MM(s + 11, &seconds))) {
410  return false;
411  }
412  return ConvertTimePoint(arrow_vendored::date::sys_days(ymd) + seconds, out);
413  }
414  if (length == 19) {
415  if (ARROW_PREDICT_FALSE(!ParseYYYY_MM_DD(s, &ymd))) {
416  return false;
417  }
418  std::chrono::duration<value_type> seconds;
419  if (ARROW_PREDICT_FALSE(!ParseHH_MM_SS(s + 11, &seconds))) {
420  return false;
421  }
422  return ConvertTimePoint(arrow_vendored::date::sys_days(ymd) + seconds, out);
423  }
424  return false;
425  }
426 
427  protected:
428  template <class TimePoint>
429  bool ConvertTimePoint(TimePoint tp, value_type* out) {
430  auto duration = tp.time_since_epoch();
431  switch (unit_) {
432  case TimeUnit::SECOND:
433  *out = std::chrono::duration_cast<std::chrono::seconds>(duration).count();
434  return true;
435  case TimeUnit::MILLI:
436  *out = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count();
437  return true;
438  case TimeUnit::MICRO:
439  *out = std::chrono::duration_cast<std::chrono::microseconds>(duration).count();
440  return true;
441  case TimeUnit::NANO:
442  *out = std::chrono::duration_cast<std::chrono::nanoseconds>(duration).count();
443  return true;
444  }
445  // Unreachable, but suppress compiler warning
446  assert(0);
447  *out = 0;
448  return true;
449  }

451  bool ParseYYYY_MM_DD(const char* s, arrow_vendored::date::year_month_day* out) {
452  uint16_t year;
453  uint8_t month, day;
454  if (ARROW_PREDICT_FALSE(s[4] != '-') || ARROW_PREDICT_FALSE(s[7] != '-')) {
455  return false;
456  }
457  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 0, 4, &year))) {
458  return false;
459  }
460  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 5, 2, &month))) {
461  return false;
462  }
463  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 8, 2, &day))) {
464  return false;
465  }
466  *out = {arrow_vendored::date::year{year}, arrow_vendored::date::month{month},
467  arrow_vendored::date::day{day}};
468  return out->ok();
469  }

471  bool ParseHH(const char* s, std::chrono::duration<value_type>* out) {
472  uint8_t hours;
473  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 0, 2, &hours))) {
474  return false;
475  }
476  if (ARROW_PREDICT_FALSE(hours >= 24)) {
477  return false;
478  }
479  *out = std::chrono::duration<value_type>(3600U * hours);
480  return true;

483  bool ParseHH_MM(const char* s, std::chrono::duration<value_type>* out) {
484  uint8_t hours, minutes;
485  if (ARROW_PREDICT_FALSE(s[2] != ':')) {
486  return false;
487  }
488  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 0, 2, &hours))) {
489  return false;
490  }
491  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 3, 2, &minutes))) {
492  return false;
493  }
494  if (ARROW_PREDICT_FALSE(hours >= 24)) {
495  return false;
496  }
497  if (ARROW_PREDICT_FALSE(minutes >= 60)) {
498  return false;
499  }
500  *out = std::chrono::duration<value_type>(3600U * hours + 60U * minutes);
501  return true;
502  }

504  bool ParseHH_MM_SS(const char* s, std::chrono::duration<value_type>* out) {
505  uint8_t hours, minutes, seconds;
506  if (ARROW_PREDICT_FALSE(s[2] != ':') || ARROW_PREDICT_FALSE(s[5] != ':')) {
507  return false;
508  }
509  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 0, 2, &hours))) {
510  return false;
511  }
512  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 3, 2, &minutes))) {
513  return false;
514  }
515  if (ARROW_PREDICT_FALSE(!detail::ParseUnsigned(s + 6, 2, &seconds))) {
516  return false;
517  }
518  if (ARROW_PREDICT_FALSE(hours >= 24)) {
519  return false;
520  }
521  if (ARROW_PREDICT_FALSE(minutes >= 60)) {
522  return false;
523  }
524  if (ARROW_PREDICT_FALSE(seconds >= 60)) {
525  return false;
526  }
527  *out = std::chrono::duration<value_type>(3600U * hours + 60U * minutes + seconds);
528  return true;
529  }
