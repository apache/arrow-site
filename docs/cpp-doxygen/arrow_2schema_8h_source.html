123  ::arrow::Status GetColumnField(int column_index, const SchemaField** out) const {
124  auto it = column_index_to_field.find(column_index);
125  if (it == column_index_to_field.end()) {
126  return ::arrow::Status::KeyError("Column index ", column_index,
127  " not found in schema manifest, may be malformed");
128  }
129  *out = it->second;
130  return ::arrow::Status::OK();

133  const SchemaField* GetParent(const SchemaField* field) const {
134  // Returns nullptr also if not found
135  auto it = child_to_parent.find(field);
136  if (it == child_to_parent.end()) {
137  return NULLPTR;
138  }
139  return it->second;
140  }

142  bool GetFieldIndices(const std::vector<int>& column_indices, std::vector<int>* out) {
143  // Coalesce a list of schema field indices which are the roots of the
144  // columns referred to by a list of column indices
145  const schema::GroupNode* group = descr->group_node();
146  std::unordered_set<int> already_added;
147  out->clear();
148  for (auto& column_idx : column_indices) {
149  auto field_node = descr->GetColumnRoot(column_idx);
150  auto field_idx = group->FieldIndex(*field_node);
151  if (field_idx < 0) {
152  return false;
153  }
154  auto insertion = already_added.insert(field_idx);
155  if (insertion.second) {
156  out->push_back(field_idx);
157  }
158  }
159  return true;
160  }
