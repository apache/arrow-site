128 // API to read values from a single column. This is a main client facing API.
129 template <typename DType>
130 class TypedColumnReader : public ColumnReader {
131  public:
132  typedef typename DType::c_type T;
133 
134  // Read a batch of repetition levels, definition levels, and values from the
135  // column.
136  //
137  // Since null values are not stored in the values, the number of values read
138  // may be less than the number of repetition and definition levels. With
139  // nested data this is almost certainly true.
140  //
141  // Set def_levels or rep_levels to nullptr if you want to skip reading them.
142  // This is only safe if you know through some other source that there are no
143  // undefined values.
144  //
145  // To fully exhaust a row group, you must read batches until the number of
146  // values read reaches the number of stored values according to the metadata.
147  //
148  // This API is the same for both V1 and V2 of the DataPage
149  //
150  // @returns: actual number of levels read (see values_read for number of values read)
151  virtual int64_t ReadBatch(int64_t batch_size, int16_t* def_levels, int16_t* rep_levels,
152  T* values, int64_t* values_read) = 0;

314 static inline void DefinitionLevelsToBitmap(
315  const int16_t* def_levels, int64_t num_def_levels, const int16_t max_definition_level,
316  const int16_t max_repetition_level, int64_t* values_read, int64_t* null_count,
317  uint8_t* valid_bits, int64_t valid_bits_offset) {
318  // We assume here that valid_bits is large enough to accommodate the
319  // additional definition levels and the ones that have already been written
320  ::arrow::internal::BitmapWriter valid_bits_writer(valid_bits, valid_bits_offset,
321  num_def_levels);
322 
323  // TODO(itaiin): As an interim solution we are splitting the code path here
324  // between repeated+flat column reads, and non-repeated+nested reads.
325  // Those paths need to be merged in the future
326  for (int i = 0; i < num_def_levels; ++i) {
327  if (def_levels[i] == max_definition_level) {
328  valid_bits_writer.Set();
329  } else if (max_repetition_level > 0) {
330  // repetition+flat case
331  if (def_levels[i] == (max_definition_level - 1)) {
332  valid_bits_writer.Clear();
333  *null_count += 1;
334  } else {
335  continue;
336  }
337  } else {
338  // non-repeated+nested case
339  if (def_levels[i] < max_definition_level) {
340  valid_bits_writer.Clear();
341  *null_count += 1;
342  } else {
343  throw ParquetException("definition level exceeds maximum");
344  }
345  }
346 
347  valid_bits_writer.Next();
348  }
349  valid_bits_writer.Finish();
350  *values_read = valid_bits_writer.position();
351 }

357 // TODO(itaiin): another code path split to merge when the general case is done
358 static inline bool HasSpacedValues(const ColumnDescriptor* descr) {
359  if (descr->max_repetition_level() > 0) {
360  // repeated+flat case
361  return !descr->schema_node()->is_required();
362  } else {
363  // non-repeated+nested case
364  // Find if a node forces nulls in the lowest level along the hierarchy
365  const schema::Node* node = descr->schema_node().get();
366  while (node) {
367  if (node->is_optional()) {
368  return true;
369  }
370  node = node->parent();
371  }
372  return false;
373  }
374 }
