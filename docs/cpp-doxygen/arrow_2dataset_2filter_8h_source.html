505 template <typename Visitor>
506 auto VisitExpression(const Expression& expr, Visitor&& visitor)
507  -> decltype(visitor(expr)) {
508  switch (expr.type()) {
509  case ExpressionType::FIELD:
510  return visitor(internal::checked_cast<const FieldExpression&>(expr));
511 
512  case ExpressionType::SCALAR:
513  return visitor(internal::checked_cast<const ScalarExpression&>(expr));
514 
515  case ExpressionType::IN:
516  return visitor(internal::checked_cast<const InExpression&>(expr));
517 
518  case ExpressionType::IS_VALID:
519  return visitor(internal::checked_cast<const IsValidExpression&>(expr));
520 
521  case ExpressionType::AND:
522  return visitor(internal::checked_cast<const AndExpression&>(expr));
523 
524  case ExpressionType::OR:
525  return visitor(internal::checked_cast<const OrExpression&>(expr));
526 
527  case ExpressionType::NOT:
528  return visitor(internal::checked_cast<const NotExpression&>(expr));
529 
530  case ExpressionType::CAST:
531  return visitor(internal::checked_cast<const CastExpression&>(expr));
532 
533  case ExpressionType::COMPARISON:
534  return visitor(internal::checked_cast<const ComparisonExpression&>(expr));
535 
536  case ExpressionType::CUSTOM:
537  default:
538  break;
539  }
540  return visitor(internal::checked_cast<const CustomExpression&>(expr));
541 }
