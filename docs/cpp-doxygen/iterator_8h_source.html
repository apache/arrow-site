364 template <typename T>
365 class FlattenIterator {
366  public:
367  explicit FlattenIterator(Iterator<Iterator<T>> it) : parent_(std::move(it)) {}
368 
369  Result<T> Next() {
370  if (child_ == IterationTraits<Iterator<T>>::End()) {
371  // Pop from parent's iterator.
372  ARROW_ASSIGN_OR_RAISE(child_, parent_.Next());
373 
374  // Check if final iteration reached.
375  if (child_ == IterationTraits<Iterator<T>>::End()) {
376  return IterationTraits<T>::End();
377  }
378 
379  return Next();
380  }
381 
382  // Pop from child_ and check for depletion.
383  ARROW_ASSIGN_OR_RAISE(T out, child_.Next());
384  if (out == IterationTraits<T>::End()) {
385  // Reset state such that we pop from parent on the recursive call
386  child_ = IterationTraits<Iterator<T>>::End();
387 
388  return Next();
389  }
390 
391  return out;
392  }
393 
394  private:
395  Iterator<Iterator<T>> parent_;
396  Iterator<T> child_ = IterationTraits<Iterator<T>>::End();
397 };

449 template <typename T>
450 class ReadaheadIterator {
451  using PromiseType = typename detail::ReadaheadIteratorPromise<T>;
452 
453  public:
454  // Public default constructor creates an empty iterator
455  ReadaheadIterator() : done_(true) {}
456 
457  ~ReadaheadIterator() {
458  if (queue_) {
459  // Make sure the queue doesn't call any promises after this object
460  // is destroyed.
461  queue_->EnsureShutdownOrDie();
462  }
463  }
464 
465  ARROW_DEFAULT_MOVE_AND_ASSIGN(ReadaheadIterator);
466  ARROW_DISALLOW_COPY_AND_ASSIGN(ReadaheadIterator);
467 
468  Result<T> Next() {
469  if (done_) {
470  return IterationTraits<T>::End();
471  }
472 
473  std::unique_ptr<detail::ReadaheadPromise> promise;
474  ARROW_RETURN_NOT_OK(queue_->PopDone(&promise));
475  auto it_promise = static_cast<PromiseType*>(promise.get());
476 
477  ARROW_RETURN_NOT_OK(queue_->Append(MakePromise()));
478 
479  ARROW_ASSIGN_OR_RAISE(auto out, std::move(it_promise->out_));
480  if (out == IterationTraits<T>::End()) {
481  done_ = true;
482  }
483  return out;
484  }
485 
486  static Result<Iterator<T>> Make(Iterator<T> it, int readahead_queue_size) {
487  ReadaheadIterator rh(std::move(it), readahead_queue_size);
488  ARROW_RETURN_NOT_OK(rh.Pump());
489  return Iterator<T>(std::move(rh));
490  }
491 
492  private:
493  explicit ReadaheadIterator(Iterator<T> it, int readahead_queue_size)
494  : it_(new Iterator<T>(std::move(it))),
495  queue_(new detail::ReadaheadQueue(readahead_queue_size)) {}
496 
497  Status Pump() {
498  return queue_->Pump([this]() { return MakePromise(); });
499  }
500 
501  std::unique_ptr<detail::ReadaheadPromise> MakePromise() {
502  return std::unique_ptr<detail::ReadaheadPromise>(new PromiseType{it_.get()});
503  }
504 
505  // The underlying iterator is referenced by pointer in ReadaheadPromise,
506  // so make sure it doesn't move.
507  std::unique_ptr<Iterator<T>> it_;
508  std::unique_ptr<detail::ReadaheadQueue> queue_;
509  bool done_ = false;
510 };
