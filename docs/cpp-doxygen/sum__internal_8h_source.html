57 template <typename ArrowType, typename StateType>
58 class SumAggregateFunction final : public AggregateFunctionStaticState<StateType> {
59  using CType = typename TypeTraits<ArrowType>::CType;
60  using ArrayType = typename TypeTraits<ArrowType>::ArrayType;
61 
62  // A small number of elements rounded to the next cacheline. This should
63  // amount to a maximum of 4 cachelines when dealing with 8 bytes elements.
64  static constexpr int64_t kTinyThreshold = 32;
65  static_assert(kTinyThreshold >= (2 * CHAR_BIT) + 1,
66  "ConsumeSparse requires 3 bytes of null bitmap, and 17 is the"
67  "required minimum number of bits/elements to cover 3 bytes.");
68 
69  public:
70  Status Consume(const Array& input, StateType* state) const override {
71  const ArrayType& array = static_cast<const ArrayType&>(input);
72 
73  if (input.null_count() == 0) {
74  *state = ConsumeDense(array);
75  } else if (input.length() <= kTinyThreshold) {
76  // In order to simplify ConsumeSparse implementation (requires at least 3
77  // bytes of bitmap data), small arrays are handled differently.
78  *state = ConsumeTiny(array);
79  } else {
80  *state = ConsumeSparse(array);
81  }
82 
83  return Status::OK();
84  }
85 
86  Status Merge(const StateType& src, StateType* dst) const override {
87  *dst += src;
88  return Status::OK();
89  }
90 
91  Status Finalize(const StateType& src, Datum* output) const override {
92  *output = src.Finalize();
93  return Status::OK();
94  }
95 
96  std::shared_ptr<DataType> out_type() const override { return StateType::out_type(); }
97 
98  private:
99  StateType ConsumeDense(const ArrayType& array) const {
100  StateType local;
101 
102  const auto values = array.raw_values();
103  const int64_t length = array.length();
104  for (int64_t i = 0; i < length; i++) {
105  local.sum += values[i];
106  }
107 
108  local.count = length;
109 
110  return local;
111  }
112 
113  StateType ConsumeTiny(const ArrayType& array) const {
114  StateType local;
115 
116  internal::BitmapReader reader(array.null_bitmap_data(), array.offset(),
117  array.length());
118  const auto values = array.raw_values();
119  for (int64_t i = 0; i < array.length(); i++) {
120  if (reader.IsSet()) {
121  local.sum += values[i];
122  local.count++;
123  }
124  reader.Next();
125  }
126 
127  return local;
128  }
129 
130  // While this is not branchless, gcc needs this to be in a different function
131  // for it to generate cmov which ends to be slightly faster than
132  // multiplication but safe for handling NaN with doubles.
133  inline CType MaskedValue(bool valid, CType value) const { return valid ? value : 0; }
134 
135  inline StateType UnrolledSum(uint8_t bits, const CType* values) const {
136  StateType local;
137 
138  if (bits < 0xFF) {
139  // Some nulls
140  for (size_t i = 0; i < 8; i++) {
141  local.sum += MaskedValue(bits & (1U << i), values[i]);
142  }
143  local.count += BitUtil::kBytePopcount[bits];
144  } else {
145  // No nulls
146  for (size_t i = 0; i < 8; i++) {
147  local.sum += values[i];
148  }
149  local.count += 8;
150  }
151 
152  return local;
153  }
154 
155  StateType ConsumeSparse(const ArrayType& array) const {
156  StateType local;
157 
158  // Sliced bitmaps on non-byte positions induce problem with the branchless
159  // unrolled technique. Thus extra padding is added on both left and right
160  // side of the slice such that both ends are byte-aligned. The first and
161  // last bitmap are properly masked to ignore extra values induced by
162  // padding.
163  //
164  // The execution is divided in 3 sections.
165  //
166  // 1. Compute the sum of the first masked byte.
167  // 2. Compute the sum of the middle bytes
168  // 3. Compute the sum of the last masked byte.
169 
170  const int64_t length = array.length();
171  const int64_t offset = array.offset();
172 
173  // The number of bytes covering the range, this includes partial bytes.
174  // This number bounded by `<= (length / 8) + 2`, e.g. a possible extra byte
175  // on the left, and on the right.
176  const int64_t covering_bytes = BitUtil::CoveringBytes(offset, length);
177  DCHECK_GE(covering_bytes, 3);
178 
179  // Align values to the first batch of 8 elements. Note that raw_values() is
180  // already adjusted with the offset, thus we rewind a little to align to
181  // the closest 8-batch offset.
182  const auto values = array.raw_values() - (offset % 8);
183 
184  // Align bitmap at the first consumable byte.
185  const auto bitmap = array.null_bitmap_data() + BitUtil::RoundDown(offset, 8) / 8;
186 
187  // Consume the first (potentially partial) byte.
188  const uint8_t first_mask = BitUtil::kTrailingBitmask[offset % 8];
189  local += UnrolledSum(bitmap[0] & first_mask, values);
190 
191  // Consume the (full) middle bytes. The loop iterates in unit of
192  // batches of 8 values and 1 byte of bitmap.
193  for (int64_t i = 1; i < covering_bytes - 1; i++) {
194  local += UnrolledSum(bitmap[i], &values[i * 8]);
195  }
196 
197  // Consume the last (potentially partial) byte.
198  const int64_t last_idx = covering_bytes - 1;
199  const uint8_t last_mask = BitUtil::kPrecedingWrappingBitmask[(offset + length) % 8];
200  local += UnrolledSum(bitmap[last_idx] & last_mask, &values[last_idx * 8]);
201 
202  return local;
203  }
