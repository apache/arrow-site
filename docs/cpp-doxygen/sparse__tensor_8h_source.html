440 
442  SparseTensorImpl(const std::shared_ptr<DataType>& type,

444  const std::vector<std::string>& dim_names = {})
445  : SparseTensorImpl(NULLPTR, type, NULLPTR, shape, dim_names) {}
446 
448  static inline Result<std::shared_ptr<SparseTensorImpl<SparseIndexType>>> Make(
449  const std::shared_ptr<SparseIndexType>& sparse_index,
450  const std::shared_ptr<DataType>& type, const std::shared_ptr<Buffer>& data,
451  const std::vector<int64_t>& shape, const std::vector<std::string>& dim_names) {
452  if (!is_tensor_supported(type->id())) {
453  return Status::Invalid(type->ToString(),
454  " is not valid data type for a sparse tensor");
455  }
456  ARROW_RETURN_NOT_OK(sparse_index->ValidateShape(shape));
457  if (dim_names.size() > 0 && dim_names.size() != shape.size()) {
458  return Status::Invalid("dim_names length is inconsistent with shape");
459  }
460  return std::make_shared<SparseTensorImpl<SparseIndexType>>(sparse_index, type, data,
461  shape, dim_names);
462  }
463 
468  static inline Result<std::shared_ptr<SparseTensorImpl<SparseIndexType>>> Make(
469  const Tensor& tensor, const std::shared_ptr<DataType>& index_value_type,
470  MemoryPool* pool = default_memory_pool()) {
471  std::shared_ptr<SparseIndex> sparse_index;
472  std::shared_ptr<Buffer> data;
473  ARROW_RETURN_NOT_OK(internal::MakeSparseTensorFromTensor(
474  tensor, SparseIndexType::format_id, index_value_type, pool, &sparse_index,
475  &data));
476  return std::make_shared<SparseTensorImpl<SparseIndexType>>(
477  internal::checked_pointer_cast<SparseIndexType>(sparse_index), tensor.type(),
478  data, tensor.shape(), tensor.dim_names_);
479  }
480 
481  static inline Result<std::shared_ptr<SparseTensorImpl<SparseIndexType>>> Make(
482  const Tensor& tensor, MemoryPool* pool = default_memory_pool()) {
