31 // The Column class is used to access data that was fetched in columnar format.
32 // The contents of the data can be accessed through the data() fn, which returns
33 // a ptr to a vector containing the contents of this column in the fetched
34 // results, avoiding copies. This vector will be of size length().
35 //
36 // If any of the values are null, they will be represented in the data vector as
37 // default values, i.e. 0 for numeric types. The nulls() fn returns a ptr to a
38 // bit array representing which values are null, and the IsNull() fn is provided
39 // for convenience when working with this bit array. The user should check
40 // IsNull() to distinguish between actual instances of the default values and nulls.
41 //
42 // A Column object is returned from a ColumnarRowSet and is only valid as long
43 // as that ColumnarRowSet still exists.
44 //
45 // Example:
46 // unique_ptr<Int32Column> col = columnar_row_set->GetInt32Col();
47 // for (int i = 0; i < col->length(); i++) {
48 // if (col->IsNull(i)) {
49 // cout << "NULL\n";
50 // } else {
51 // cout << col->data()[i] << "\n";
52 // }
53 // }
54 class ARROW_EXPORT Column {
55  public:
56  virtual ~Column() {}
57 
58  virtual int64_t length() const = 0;
59 
60  const uint8_t* nulls() const { return nulls_; }
61  int64_t nulls_size() const { return nulls_size_; }
62 
63  // Returns true iff the value for the i-th row within this set of data for this
64  // column is null.
65  bool IsNull(int64_t i) const { return (nulls_[i / 8] & (1 << (i % 8))) != 0; }
66 
67  protected:
68  explicit Column(const std::string* nulls);
69 
70  // The memory for these ptrs is owned by the ColumnarRowSet that
71  // created this Column.
72  //
73  // Due to the issue described in HUE-2722, the null bitmap may have fewer
74  // bytes than expected for some versions of Hive, so we retain the ability to
75  // check the buffer size in case this happens.
76  const uint8_t* nulls_;
77  int64_t nulls_size_;
78 };

108 // A ColumnarRowSet represents the full results returned by a call to
109 // Operation::Fetch() when a columnar format is being used.
110 //
111 // ColumnarRowSet provides access to specific columns by their type and index in
112 // the results. All Column objects returned from a given ColumnarRowSet will have
113 // the same length(). A Column object returned by a ColumnarRowSet is only valid
114 // as long as the ColumnarRowSet still exists.
