99  template <typename Visitor>
100  Status VisitColumn(int32_t col_index, Visitor&& visit) const {
101  for (size_t buf_index = 0; buf_index < values_buffers_.size(); ++buf_index) {
102  const auto& values_buffer = values_buffers_[buf_index];
103  const auto values = reinterpret_cast<const ValueDesc*>(values_buffer->data());
104  const auto max_pos =
105  static_cast<int32_t>(values_buffer->size() / sizeof(ValueDesc)) - 1;
106  for (int32_t pos = col_index; pos < max_pos; pos += num_cols_) {
107  auto start = values[pos].offset;
108  auto stop = values[pos + 1].offset;
109  auto quoted = values[pos + 1].quoted;
110  ARROW_RETURN_NOT_OK(visit(parsed_ + start, stop - start, quoted));
111  }
112  }
113  return Status::OK();
114  }

116  template <typename Visitor>
117  Status VisitLastRow(Visitor&& visit) const {
118  const auto& values_buffer = values_buffers_.back();
119  const auto values = reinterpret_cast<const ValueDesc*>(values_buffer->data());
120  const auto start_pos =
121  static_cast<int32_t>(values_buffer->size() / sizeof(ValueDesc)) - num_cols_ - 1;
122  for (int32_t col_index = 0; col_index < num_cols_; ++col_index) {
123  auto start = values[start_pos + col_index].offset;
124  auto stop = values[start_pos + col_index + 1].offset;
125  auto quoted = values[start_pos + col_index + 1].quoted;
126  ARROW_RETURN_NOT_OK(visit(parsed_ + start, stop - start, quoted));
127  }
128  return Status::OK();
129  }
