534 // Julian day at unix epoch.
535 //
536 // The Julian Day Number (JDN) is the integer assigned to a whole solar day in
537 // the Julian day count starting from noon Universal time, with Julian day
538 // number 0 assigned to the day starting at noon on Monday, January 1, 4713 BC,
539 // proleptic Julian calendar (November 24, 4714 BC, in the proleptic Gregorian
540 // calendar),
541 constexpr int64_t kJulianToUnixEpochDays = INT64_C(2440588);
542 constexpr int64_t kSecondsPerDay = INT64_C(60 * 60 * 24);
543 constexpr int64_t kMillisecondsPerDay = kSecondsPerDay * INT64_C(1000);
544 constexpr int64_t kMicrosecondsPerDay = kMillisecondsPerDay * INT64_C(1000);
545 constexpr int64_t kNanosecondsPerDay = kMicrosecondsPerDay * INT64_C(1000);
546 
547 MANUALLY_ALIGNED_STRUCT(1) Int96 { uint32_t value[3]; };
548 STRUCT_END(Int96, 12);
549 
550 inline bool operator==(const Int96& left, const Int96& right) {
551  return std::equal(left.value, left.value + 3, right.value);
552 }

554 inline bool operator!=(const Int96& left, const Int96& right) { return !(left == right); }

560 static inline void Int96SetNanoSeconds(parquet::Int96& i96, int64_t nanoseconds) {
561  std::memcpy(&i96.value, &nanoseconds, sizeof(nanoseconds));

564 static inline int64_t Int96GetNanoSeconds(const parquet::Int96& i96) {
565  int64_t days_since_epoch = i96.value[2] - kJulianToUnixEpochDays;
566  int64_t nanoseconds = 0;
567 
568  memcpy(&nanoseconds, &i96.value, sizeof(int64_t));
569  return days_since_epoch * kNanosecondsPerDay + nanoseconds;
570 }
