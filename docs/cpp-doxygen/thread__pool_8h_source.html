61 // Needed because std::packaged_task is not copyable and hence not convertible
62 // to std::function.
63 template <typename R, typename... Args>
64 struct packaged_task_wrapper {
65  using PackagedTask = std::packaged_task<R(Args...)>;
66 
67  explicit packaged_task_wrapper(PackagedTask&& task)
68  : task_(std::make_shared<PackagedTask>(std::forward<PackagedTask>(task))) {}
69 
70  void operator()(Args&&... args) { return (*task_)(std::forward<Args>(args)...); }
71  std::shared_ptr<PackagedTask> task_;
72 };

111  // Submit a callable and arguments for execution. Return a future that
112  // will return the callable's result value once.
113  // The callable's arguments are copied before execution.
114  template <typename Function, typename... Args,
115  typename RetType = typename std::result_of<Function && (Args && ...)>::type>
116  Result<std::future<RetType>> Submit(Function&& func, Args&&... args) {
117  // Trying to templatize std::packaged_task with Function doesn't seem
118  // to work, so go through std::bind to simplify the packaged signature
119  using PackagedTask = std::packaged_task<RetType()>;
120  auto task = PackagedTask(
121  std::bind(std::forward<Function>(func), std::forward<Args>(args)...));
122  auto fut = task.get_future();
123 
124  ARROW_RETURN_NOT_OK(
125  SpawnReal(detail::packaged_task_wrapper<RetType>(std::move(task))));
126  return std::move(fut);
127  }
