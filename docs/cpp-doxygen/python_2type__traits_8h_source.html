311 static inline int NumPyTypeSize(int npy_type) {
312  npy_type = fix_numpy_type_num(npy_type);
313 
314  switch (npy_type) {

316  case NPY_INT8:
317  case NPY_UINT8:

319  case NPY_INT16:
320  case NPY_UINT16:
321  return 2;
322  case NPY_INT32:
323  case NPY_UINT32:
324  return 4;
325  case NPY_INT64:
326  case NPY_UINT64:
327  return 8;
328  case NPY_FLOAT16:
329  return 2;
330  case NPY_FLOAT32:
331  return 4;
332  case NPY_FLOAT64:
333  return 8;

335  return 8;
336  case NPY_OBJECT:
337  return sizeof(void*);
338  default:
339  ARROW_CHECK(false) << "unhandled numpy type";
