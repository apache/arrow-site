402 template <typename Range>
403 Status TableFromTupleRange(MemoryPool* pool, Range&& rows,
404  const std::vector<std::string>& names,
405  std::shared_ptr<Table>* table) {
406  using row_type = typename std::iterator_traits<decltype(std::begin(rows))>::value_type;
407  constexpr std::size_t n_columns = std::tuple_size<row_type>::value;
408 
409  std::shared_ptr<Schema> schema = SchemaFromTuple<row_type>::MakeSchema(names);
410 
411  std::vector<std::unique_ptr<ArrayBuilder>> builders(n_columns);
412  ARROW_RETURN_NOT_OK(internal::CreateBuildersRecursive<row_type>::Make(pool, &builders));
413 
414  for (auto const& row : rows) {
415  ARROW_RETURN_NOT_OK(internal::RowIterator<row_type>::Append(builders, row));
416  }
417 
418  std::vector<std::shared_ptr<Array>> arrays;
419  for (auto const& builder : builders) {
420  std::shared_ptr<Array> array;
421  ARROW_RETURN_NOT_OK(builder->Finish(&array));
422  arrays.emplace_back(array);
423  }
424 
425  *table = Table::Make(schema, arrays);
426 
427  return Status::OK();
428 }

430 template <typename Range>
431 Status TupleRangeFromTable(const Table& table, const compute::CastOptions& cast_options,
432  compute::FunctionContext* ctx, Range* rows) {
433  using row_type = typename std::decay<decltype(*std::begin(*rows))>::type;
434  constexpr std::size_t n_columns = std::tuple_size<row_type>::value;
435 
436  if (table.schema()->num_fields() != n_columns) {
437  std::stringstream ss;
438  ss << "Number of columns in the table does not match the width of the target: ";
439  ss << table.schema()->num_fields() << " != " << n_columns;
440  return Status::Invalid(ss.str());
441  }
442 
443  // TODO: Use std::size with C++17
444  if (rows->size() != static_cast<size_t>(table.num_rows())) {
445  std::stringstream ss;
446  ss << "Number of rows in the table does not match the size of the target: ";
447  ss << table.num_rows() << " != " << rows->size();
448  return Status::Invalid(ss.str());
449  }
450 
451  // Check that all columns have the correct type, otherwise cast them.
452  std::shared_ptr<Table> table_owner;
453  std::reference_wrapper<const ::arrow::Table> current_table(table);
454 
455  ARROW_RETURN_NOT_OK(internal::EnsureColumnTypes<row_type>::Cast(
456  table, &table_owner, cast_options, ctx, &current_table));
457 
458  internal::TupleSetter<Range, row_type>::Fill(current_table.get(), rows);
459 
460  return Status::OK();
461 }

532 
533  Status Allocate(int64_t size, uint8_t** out) override {
534  try {
535  *out = alloc_.allocate(size);
536  } catch (std::bad_alloc& e) {
537  return Status::OutOfMemory(e.what());
538  }
539  stats_.UpdateAllocatedBytes(size);
540  return Status::OK();

542 
543  Status Reallocate(int64_t old_size, int64_t new_size, uint8_t** ptr) override {
544  uint8_t* old_ptr = *ptr;
545  try {
546  *ptr = alloc_.allocate(new_size);
547  } catch (std::bad_alloc& e) {
548  return Status::OutOfMemory(e.what());
549  }
550  memcpy(*ptr, old_ptr, std::min(old_size, new_size));
551  alloc_.deallocate(old_ptr, old_size);
552  stats_.UpdateAllocatedBytes(new_size - old_size);
553  return Status::OK();
554  }

557  alloc_.deallocate(buffer, size);
558  stats_.UpdateAllocatedBytes(-size);
559  }

562 
563  int64_t max_memory() const override { return stats_.max_memory(); }

565  std::string backend_name() const override { return "stl"; }
