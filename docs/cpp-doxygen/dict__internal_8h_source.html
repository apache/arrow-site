59 template <>
60 struct DictionaryTraits<BooleanType> {
61  using T = BooleanType;
62  using MemoTableType = typename HashTraits<T>::MemoTableType;
63 
64  static Status GetDictionaryArrayData(MemoryPool* pool,
65  const std::shared_ptr<DataType>& type,
66  const MemoTableType& memo_table,
67  int64_t start_offset,
68  std::shared_ptr<ArrayData>* out) {
69  if (start_offset < 0) {
70  return Status::Invalid("invalid start_offset ", start_offset);
71  }
72 
73  BooleanBuilder builder(pool);
74  const auto& bool_values = memo_table.values();
75  const auto null_index = memo_table.GetNull();
76 
77  // Will iterate up to 3 times.
78  for (int64_t i = start_offset; i < memo_table.size(); i++) {
79  RETURN_NOT_OK(i == null_index ? builder.AppendNull()
80  : builder.Append(bool_values[i]));
81  }
82 
83  return builder.FinishInternal(out);
84  }
85 }; // namespace internal

87 template <typename T>
88 struct DictionaryTraits<T, enable_if_has_c_type<T>> {
89  using c_type = typename T::c_type;
90  using MemoTableType = typename HashTraits<T>::MemoTableType;
91 
92  static Status GetDictionaryArrayData(MemoryPool* pool,
93  const std::shared_ptr<DataType>& type,
94  const MemoTableType& memo_table,
95  int64_t start_offset,
96  std::shared_ptr<ArrayData>* out) {
97  std::shared_ptr<Buffer> dict_buffer;
98  auto dict_length = static_cast<int64_t>(memo_table.size()) - start_offset;
99  // This makes a copy, but we assume a dictionary array is usually small
100  // compared to the size of the dictionary-using array.
101  // (also, copying the dictionary values is cheap compared to the cost
102  // of building the memo table)
103  RETURN_NOT_OK(
104  AllocateBuffer(pool, TypeTraits<T>::bytes_required(dict_length), &dict_buffer));
105  memo_table.CopyValues(static_cast<int32_t>(start_offset),
106  reinterpret_cast<c_type*>(dict_buffer->mutable_data()));
107 
108  int64_t null_count = 0;
109  std::shared_ptr<Buffer> null_bitmap = nullptr;
110  RETURN_NOT_OK(
111  ComputeNullBitmap(pool, memo_table, start_offset, &null_count, &null_bitmap));
112 
113  *out = ArrayData::Make(type, dict_length, {null_bitmap, dict_buffer}, null_count);
114  return Status::OK();
115  }
116 };

118 template <typename T>
119 struct DictionaryTraits<T, enable_if_base_binary<T>> {
120  using MemoTableType = typename HashTraits<T>::MemoTableType;
121 
122  static Status GetDictionaryArrayData(MemoryPool* pool,
123  const std::shared_ptr<DataType>& type,
124  const MemoTableType& memo_table,
125  int64_t start_offset,
126  std::shared_ptr<ArrayData>* out) {
127  std::shared_ptr<Buffer> dict_offsets;
128  std::shared_ptr<Buffer> dict_data;
129 
130  // Create the offsets buffer
131  auto dict_length = static_cast<int64_t>(memo_table.size() - start_offset);
132  if (dict_length > 0) {
133  RETURN_NOT_OK(AllocateBuffer(
134  pool, TypeTraits<Int32Type>::bytes_required(dict_length + 1), &dict_offsets));
135  auto raw_offsets = reinterpret_cast<int32_t*>(dict_offsets->mutable_data());
136  memo_table.CopyOffsets(static_cast<int32_t>(start_offset), raw_offsets);
137  }
138 
139  // Create the data buffer
140  auto values_size = memo_table.values_size();
141  if (values_size > 0) {
142  RETURN_NOT_OK(AllocateBuffer(pool, values_size, &dict_data));
143  memo_table.CopyValues(static_cast<int32_t>(start_offset), dict_data->size(),
144  dict_data->mutable_data());
145  }
146 
147  int64_t null_count = 0;
148  std::shared_ptr<Buffer> null_bitmap = nullptr;
149  RETURN_NOT_OK(
150  ComputeNullBitmap(pool, memo_table, start_offset, &null_count, &null_bitmap));
151 
152  *out = ArrayData::Make(type, dict_length, {null_bitmap, dict_offsets, dict_data},
153  null_count);
154 
155  return Status::OK();
156  }

159 template <typename T>
160 struct DictionaryTraits<T, enable_if_fixed_size_binary<T>> {
161  using MemoTableType = typename HashTraits<T>::MemoTableType;
162 
163  static Status GetDictionaryArrayData(MemoryPool* pool,
164  const std::shared_ptr<DataType>& type,
165  const MemoTableType& memo_table,
166  int64_t start_offset,
167  std::shared_ptr<ArrayData>* out) {
168  const T& concrete_type = internal::checked_cast<const T&>(*type);
169  std::shared_ptr<Buffer> dict_data;
170 
171  // Create the data buffer
172  auto dict_length = static_cast<int64_t>(memo_table.size() - start_offset);
173  auto width_length = concrete_type.byte_width();
174  auto data_length = dict_length * width_length;
175  RETURN_NOT_OK(AllocateBuffer(pool, data_length, &dict_data));
176  auto data = dict_data->mutable_data();
177 
178  memo_table.CopyFixedWidthValues(static_cast<int32_t>(start_offset), width_length,
179  data_length, data);
180 
181  int64_t null_count = 0;
182  std::shared_ptr<Buffer> null_bitmap = nullptr;
183  RETURN_NOT_OK(
184  ComputeNullBitmap(pool, memo_table, start_offset, &null_count, &null_bitmap));
185 
186  *out = ArrayData::Make(type, dict_length, {null_bitmap, dict_data}, null_count);
187  return Status::OK();
188  }
