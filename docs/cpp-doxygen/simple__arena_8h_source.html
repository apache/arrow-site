103 inline uint8_t* SimpleArena::Allocate(int64_t size) {
104  if (avail_bytes_ < size) {
105  auto status = AllocateChunk(std::max(size, min_chunk_size_));
106  if (!status.ok()) {
107  return NULL;
108  }
109  }
110 
111  uint8_t* ret = avail_buf_;
112  avail_buf_ += size;
113  avail_bytes_ -= size;
114  return ret;
115 }

117 inline arrow::Status SimpleArena::AllocateChunk(int64_t size) {
118  uint8_t* out;
119 
120  auto status = pool_->Allocate(size, &out);
121  ARROW_RETURN_NOT_OK(status);
122 
123  chunks_.emplace_back(out, size);
124  avail_buf_ = out;
125  avail_bytes_ = size; // left-over bytes in the previous chunk cannot be used anymore.
126  total_bytes_ += size;
127  return arrow::Status::OK();
128 }

130 // In the most common case, a chunk will be allocated when processing the first record.
131 // And, the same chunk can be used for processing the remaining records in the batch.
132 // By retaining the first chunk, the number of malloc calls are reduced to one per batch,
133 // instead of one per record.
134 inline void SimpleArena::Reset() {
135  if (chunks_.size() == 0) {
136  // if there are no chunks, nothing to do.
137  return;
138  }
139 
140  // Release all but the first chunk.
141  if (chunks_.size() > 1) {
142  ReleaseChunks(true);
143  chunks_.erase(chunks_.begin() + 1, chunks_.end());
144  }
145 
146  avail_buf_ = chunks_.at(0).buf_;
147  avail_bytes_ = total_bytes_ = chunks_.at(0).size_;
148 }
149 
150 inline void SimpleArena::ReleaseChunks(bool retain_first) {
151  for (auto& chunk : chunks_) {
152  if (retain_first) {
153  // skip freeing first chunk.
154  retain_first = false;
155  continue;
156  }
157  pool_->Free(chunk.buf_, chunk.size_);
158  }
159 }
