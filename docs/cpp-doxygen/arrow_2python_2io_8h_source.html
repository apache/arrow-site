82 // A Buffer subclass that keeps a PyObject reference throughout its
83 // lifetime, such that the Python object is kept alive as long as the
84 // C++ buffer is still needed.
85 // Keeping the reference in a Python wrapper would be incorrect as
86 // the Python wrapper can get destroyed even though the wrapped C++
87 // buffer is still alive (ARROW-2270).
88 class ARROW_PYTHON_EXPORT PyForeignBuffer : public Buffer {
89  public:
90  static Status Make(const uint8_t* data, int64_t size, PyObject* base,
91  std::shared_ptr<Buffer>* out);
92 
93  private:
94  PyForeignBuffer(const uint8_t* data, int64_t size, PyObject* base)
95  : Buffer(data, size) {
96  Py_INCREF(base);
97  base_.reset(base);
98  }
99 
100  OwnedRefNoGIL base_;
101 };
