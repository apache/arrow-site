190  template <typename T1 = T>
191  enable_if_t<!is_fixed_size_binary_type<T1>::value, Status> AppendArray(
192  const Array& array) {
193  using ArrayType = typename TypeTraits<T>::ArrayType;
194 
195  const auto& concrete_array = static_cast<const ArrayType&>(array);
196  for (int64_t i = 0; i < array.length(); i++) {
197  if (array.IsNull(i)) {
198  ARROW_RETURN_NOT_OK(AppendNull());
199  } else {
200  ARROW_RETURN_NOT_OK(Append(concrete_array.GetView(i)));
201  }
202  }
203  return Status::OK();
204  }

206  template <typename T1 = T>
207  enable_if_fixed_size_binary<T1, Status> AppendArray(const Array& array) {
208  if (!value_type_->Equals(*array.type())) {
209  return Status::Invalid(
210  "Cannot append FixedSizeBinary array with non-matching type");
211  }
212 
213  const auto& concrete_array = static_cast<const FixedSizeBinaryArray&>(array);
214  for (int64_t i = 0; i < array.length(); i++) {
215  if (array.IsNull(i)) {
216  ARROW_RETURN_NOT_OK(AppendNull());
217  } else {
218  ARROW_RETURN_NOT_OK(Append(concrete_array.GetValue(i)));
219  }
220  }
221  return Status::OK();
222  }

277  Status FinishWithDictOffset(int64_t dict_offset,
278  std::shared_ptr<ArrayData>* out_indices,
279  std::shared_ptr<Array>* out_dictionary) {
280  // Finalize indices array
281  ARROW_RETURN_NOT_OK(indices_builder_.FinishInternal(out_indices));
282 
283  // Generate dictionary array from hash table contents
284  std::shared_ptr<ArrayData> dictionary_data;
285  ARROW_RETURN_NOT_OK(memo_table_->GetArrayData(pool_, dict_offset, &dictionary_data));
286 
287  *out_dictionary = MakeArray(dictionary_data);
288  delta_offset_ = memo_table_->size();
289 
290  // Update internals for further uses of this DictionaryBuilder
291  ArrayBuilder::Reset();
292  return Status::OK();
293  }

300 
301  // Only used for FixedSizeBinaryType
302  int32_t byte_width_;
303 
304  BuilderType indices_builder_;
305  std::shared_ptr<DataType> value_type_;

389  Status AppendIndices(const int64_t* values, int64_t length,
390  const uint8_t* valid_bytes = NULLPTR) {
391  int64_t null_count_before = this->indices_builder_.null_count();
392  ARROW_RETURN_NOT_OK(this->indices_builder_.AppendValues(values, length, valid_bytes));
393  this->capacity_ = this->indices_builder_.capacity();
394  this->length_ += length;
395  this->null_count_ += this->indices_builder_.null_count() - null_count_before;
396  return Status::OK();
