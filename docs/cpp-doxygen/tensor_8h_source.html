148  static int64_t CalculateValueOffset(const std::vector<int64_t>& strides,
149  const std::vector<int64_t>& index) {
150  const int64_t n = static_cast<int64_t>(index.size());
151  int64_t offset = 0;
152  for (int64_t i = 0; i < n; ++i) {
153  offset += index[i] * strides[i];
154  }
155  return offset;
156  }
157 
159  template <typename ValueType>
160  const typename ValueType::c_type& Value(const std::vector<int64_t>& index) const {
161  using c_type = typename ValueType::c_type;
162  const int64_t offset = Tensor::CalculateValueOffset(strides_, index);
163  const c_type* ptr = reinterpret_cast<const c_type*>(raw_data() + offset);
164  return *ptr;
165  }
