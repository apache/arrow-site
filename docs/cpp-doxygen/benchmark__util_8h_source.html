48 // Benchmark changed its parameter type between releases from

54 void BenchmarkSetArgsWithSizes(benchmark::internal::Benchmark* bench,
55  const std::vector<int64_t>& sizes = kMemorySizes) {
56  bench->Unit(benchmark::kMicrosecond);
57 
58  for (auto size : sizes)
59  for (auto nulls : std::vector<ArgsType>({0, 1, 10, 50}))
60  bench->Args({static_cast<ArgsType>(size), nulls});
61 }

63 void BenchmarkSetArgs(benchmark::internal::Benchmark* bench) {
