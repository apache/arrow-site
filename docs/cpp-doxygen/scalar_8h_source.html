401 template <typename ValueRef>
402 struct MakeScalarImpl {
403  template <
404  typename T, typename ScalarType = typename TypeTraits<T>::ScalarType,
405  typename ValueType = typename ScalarType::ValueType,
406  typename Enable = typename std::enable_if<
407  internal::is_simple_scalar<ScalarType>::value &&
408  std::is_same<ValueType, typename std::decay<ValueRef>::type>::value>::type>
409  Status Visit(const T& t) {
410  ARROW_RETURN_NOT_OK(internal::CheckBufferLength(&t, &value_));
411  *out_ = std::make_shared<ScalarType>(ValueType(static_cast<ValueRef>(value_)), type_);
412  return Status::OK();
413  }
414 
415  Status Visit(const DataType& t) {
416  return Status::NotImplemented("constructing scalars of type ", t, " from ", value_);
417  }
418 
419  const std::shared_ptr<DataType>& type_;
420  ValueRef value_;
421  std::shared_ptr<Scalar>* out_;
422 };

424 template <typename Value>
425 Result<std::shared_ptr<Scalar>> MakeScalar(const std::shared_ptr<DataType>& type,
426  Value&& value) {
427  std::shared_ptr<Scalar> out;
