52  // The maximum scale representable.

55  // When operating on decimal inputs, the integer part of the output can exceed the
56  // max precision. In such cases, the scale can be reduced, upto a minimum of
57  // kMinAdjustedScale.
58  // * There is no strong reason for 6, but both SQLServer and Impala use 6 too.
59  static constexpr int32_t kMinAdjustedScale = 6;
60 
61  // For specified operation and input scale/precision, determine the output
62  // scale/precision.
63  static Status GetResultType(Op op, const Decimal128TypeVector& in_types,
64  Decimal128TypePtr* out_type);

66  static Decimal128TypePtr MakeType(int32_t precision, int32_t scale);
67 
68  private:
69  // Reduce the scale if possible so that precision stays <= kMaxPrecision
70  static Decimal128TypePtr MakeAdjustedType(int32_t precision, int32_t scale) {
71  if (precision > kMaxPrecision) {
72  int32_t min_scale = std::min(scale, kMinAdjustedScale);
73  int32_t delta = precision - kMaxPrecision;
74  precision = kMaxPrecision;
75  scale = std::max(scale - delta, min_scale);
76  }
77  return MakeType(precision, scale);
78  }
79 };
