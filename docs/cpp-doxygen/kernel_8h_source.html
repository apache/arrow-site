223  bool Equals(const Datum& other) const {
224  if (this->kind() != other.kind()) return false;
225 
226  switch (this->kind()) {
227  case Datum::NONE:
228  return true;
229  case Datum::SCALAR:
230  return internal::SharedPtrEquals(this->scalar(), other.scalar());
231  case Datum::ARRAY:
232  return internal::SharedPtrEquals(this->make_array(), other.make_array());
233  case Datum::CHUNKED_ARRAY:
234  return internal::SharedPtrEquals(this->chunked_array(), other.chunked_array());
235  case Datum::RECORD_BATCH:
236  return internal::SharedPtrEquals(this->record_batch(), other.record_batch());
237  case Datum::TABLE:
238  return internal::SharedPtrEquals(this->table(), other.table());
239  case Datum::COLLECTION:
240  return CollectionEquals(this->collection(), other.collection());
241  default:
242  return false;
243  }

277 static inline bool CollectionEquals(const std::vector<Datum>& left,
278  const std::vector<Datum>& right) {
279  if (left.size() != right.size()) return false;
280 
281  for (size_t i = 0; i < left.size(); i++)
282  if (!left[i].Equals(right[i])) return false;
283 
284  return true;
285 }
