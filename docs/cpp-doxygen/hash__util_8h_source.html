77 #ifdef ARROW_HAVE_ARMV8_CRYPTO
78 /* Crc32c Parallel computation
79  * Algorithm comes from Intel whitepaper:
80  * crc-iscsi-polynomial-crc32-instruction-paper
81  *
82  * Input data is divided into three equal-sized blocks
83  * Three parallel blocks (crc0, crc1, crc2) for 1024 Bytes
84  * One Block: 42(BLK_LENGTH) * 8(step length: crc32c_u64) bytes
85  */
86 #define BLK_LENGTH 42
87  static uint32_t Armv8CrcHashParallel(const void* data, int32_t nbytes, uint32_t crc) {
88  const uint8_t* buf8;
89  const uint64_t* buf64 = reinterpret_cast<const uint64_t*>(data);
90  int32_t length = nbytes;
91 
92  while (length >= 1024) {
93  uint64_t t0, t1;
94  uint32_t crc0 = 0, crc1 = 0, crc2 = 0;
95 
96  /* parallel computation params:
97  * k0 = CRC32(x ^ (42 * 8 * 8 * 2 - 1));
98  * k1 = CRC32(x ^ (42 * 8 * 8 - 1));
99  */
100  uint32_t k0 = 0xe417f38a, k1 = 0x8f158014;
101 
102  /* First 8 byte for better pipelining */
103  crc0 = ARMCE_crc32_u64(crc, *buf64++);
104 
105  /* 3 blocks crc32c parallel computation
106  *
107  * 42 * 8 * 3 = 1008 (bytes)
108  */
109  for (int i = 0; i < BLK_LENGTH; i++, buf64++) {
110  crc0 = ARMCE_crc32_u64(crc0, *buf64);
111  crc1 = ARMCE_crc32_u64(crc1, *(buf64 + BLK_LENGTH));
112  crc2 = ARMCE_crc32_u64(crc2, *(buf64 + (BLK_LENGTH * 2)));
113  }
114  buf64 += (BLK_LENGTH * 2);
115 
116  /* Last 8 bytes */
117  crc = ARMCE_crc32_u64(crc2, *buf64++);
118 
119  t0 = (uint64_t)vmull_p64(crc0, k0);
120  t1 = (uint64_t)vmull_p64(crc1, k1);
121 
122  /* Merge (crc0, crc1, crc2) -> crc */
123  crc1 = ARMCE_crc32_u64(0, t1);
124  crc ^= crc1;
125  crc0 = ARMCE_crc32_u64(0, t0);
126  crc ^= crc0;
127 
128  length -= 1024;
129  }
130 
131  buf8 = reinterpret_cast<const uint8_t*>(buf64);
132  while (length >= 8) {
133  crc = ARMCE_crc32_u64(crc, *reinterpret_cast<const uint64_t*>(buf8));
134  buf8 += 8;
135  length -= 8;
136  }
137 
138  /* The following is more efficient than the straight loop */
139  if (length >= 4) {
140  crc = ARMCE_crc32_u32(crc, *reinterpret_cast<const uint32_t*>(buf8));
141  buf8 += 4;
142  length -= 4;
143  }
144 
145  if (length >= 2) {
146  crc = ARMCE_crc32_u16(crc, *reinterpret_cast<const uint16_t*>(buf8));
147  buf8 += 2;
148  length -= 2;
149  }
150 
151  if (length >= 1) crc = ARMCE_crc32_u8(crc, *(buf8));
152 
153  return crc;
154  }
155 #endif

163  static uint32_t CrcHash(const void* data, int32_t nbytes, uint32_t hash) {
164  const uint8_t* p = reinterpret_cast<const uint8_t*>(data);
165  const uint8_t* end = p + nbytes;
166 
167 #if ARROW_BITNESS >= 64
168  while (p <= end - 8) {
169  hash = HW_crc32_u64(hash, *reinterpret_cast<const uint64_t*>(p));
170  p += 8;
171  }
172 #endif
173 
174  while (p <= end - 4) {
175  hash = HW_crc32_u32(hash, *reinterpret_cast<const uint32_t*>(p));
176  p += 4;
177  }
178  while (p < end) {
179  hash = HW_crc32_u8(hash, *p);
180  ++p;
181  }
182 
183  // The lower half of the CRC hash has has poor uniformity, so swap the halves
184  // for anyone who only uses the first several bits of the hash.
185  hash = (hash << 16) | (hash >> 16);
186  return hash;
187  }

196  static uint64_t DoubleCrcHash(const void* data, int32_t nbytes, uint64_t hash) {
197  const uint8_t* p = reinterpret_cast<const uint8_t*>(data);
198 
199  uint32_t h1 = static_cast<uint32_t>(hash >> 32);
200  uint32_t h2 = static_cast<uint32_t>(hash);
201 
202 #if ARROW_BITNESS >= 64
203  while (nbytes >= 16) {
204  h1 = HW_crc32_u64(h1, *reinterpret_cast<const uint64_t*>(p));
205  h2 = HW_crc32_u64(h2, *reinterpret_cast<const uint64_t*>(p + 8));
206  nbytes -= 16;
207  p += 16;
208  }
209  if (nbytes >= 8) {
210  h1 = HW_crc32_u32(h1, *reinterpret_cast<const uint32_t*>(p));
211  h2 = HW_crc32_u32(h2, *reinterpret_cast<const uint32_t*>(p + 4));
212  nbytes -= 8;
213  p += 8;
214  }
215 #else
216  while (nbytes >= 8) {
217  h1 = HW_crc32_u32(h1, *reinterpret_cast<const uint32_t*>(p));
218  h2 = HW_crc32_u32(h2, *reinterpret_cast<const uint32_t*>(p + 4));
219  nbytes -= 8;
220  p += 8;
221  }
222 #endif
223 
224  if (nbytes >= 4) {
225  h1 = HW_crc32_u16(h1, *reinterpret_cast<const uint16_t*>(p));
226  h2 = HW_crc32_u16(h2, *reinterpret_cast<const uint16_t*>(p + 2));
227  nbytes -= 4;
228  p += 4;
229  }
230  switch (nbytes) {
231  case 3:
232  h1 = HW_crc32_u8(h1, p[2]);
233  // fallthrough
234  case 2:
235  h2 = HW_crc32_u8(h2, p[1]);
236  // fallthrough
237  case 1:
238  h1 = HW_crc32_u8(h1, p[0]);
239  // fallthrough
240  case 0:
241  break;
242  default:
243  assert(0);
244  }
245 
246  // A finalization step is recommended to mix up the result's bits
247  return (static_cast<uint64_t>(h1) << 32) + h2;
248  }
249 };
