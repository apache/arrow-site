52  Status Resize(int64_t capacity) override {
53  if (capacity > maximum_elements()) {
54  return Status::CapacityError("List array cannot reserve space for more than ",
55  maximum_elements(), " got ", capacity);
56  }
57  ARROW_RETURN_NOT_OK(CheckCapacity(capacity, capacity_));
58 
59  // one more then requested for offsets
60  ARROW_RETURN_NOT_OK(offsets_builder_.Resize(capacity + 1));
61  return ArrayBuilder::Resize(capacity);
62  }
63 
64  void Reset() override {
65  ArrayBuilder::Reset();
66  offsets_builder_.Reset();
67  value_builder_->Reset();
68  }
69 
74  Status AppendValues(const offset_type* offsets, int64_t length,
75  const uint8_t* valid_bytes = NULLPTR) {
76  ARROW_RETURN_NOT_OK(Reserve(length));
77  UnsafeAppendToBitmap(valid_bytes, length);
78  offsets_builder_.UnsafeAppend(offsets, length);
79  return Status::OK();
80  }
81 
86  Status Append(bool is_valid = true) {
87  ARROW_RETURN_NOT_OK(Reserve(1));
88  UnsafeAppendToBitmap(is_valid);
89  return AppendNextOffset();
90  }
91 
92  Status AppendNull() final { return Append(false); }
93 
94  Status AppendNulls(int64_t length) final {
95  ARROW_RETURN_NOT_OK(Reserve(length));
96  ARROW_RETURN_NOT_OK(CheckNextOffset());
97  UnsafeAppendToBitmap(length, false);
98  const int64_t num_values = value_builder_->length();
99  for (int64_t i = 0; i < length; ++i) {
100  offsets_builder_.UnsafeAppend(static_cast<offset_type>(num_values));
101  }
102  return Status::OK();
103  }
104 
105  Status FinishInternal(std::shared_ptr<ArrayData>* out) override {
106  ARROW_RETURN_NOT_OK(AppendNextOffset());
107 
108  // Offset padding zeroed by BufferBuilder
109  std::shared_ptr<Buffer> offsets, null_bitmap;
110  ARROW_RETURN_NOT_OK(offsets_builder_.Finish(&offsets));
111  ARROW_RETURN_NOT_OK(null_bitmap_builder_.Finish(&null_bitmap));
112 
113  if (value_builder_->length() == 0) {
114  // Try to make sure we get a non-null values buffer (ARROW-2744)
115  ARROW_RETURN_NOT_OK(value_builder_->Resize(0));
116  }
117 
118  std::shared_ptr<ArrayData> items;
119  ARROW_RETURN_NOT_OK(value_builder_->FinishInternal(&items));
120 
121  *out = ArrayData::Make(type(), length_, {null_bitmap, offsets}, {std::move(items)},
122  null_count_);
123  Reset();
124  return Status::OK();
125  }
126 
127  ArrayBuilder* value_builder() const { return value_builder_.get(); }
128 
129  // Cannot make this a static attribute because of linking issues
130  static constexpr int64_t maximum_elements() {
131  return std::numeric_limits<offset_type>::max() - 1;
132  }
133 
134  std::shared_ptr<DataType> type() const override {
135  return std::make_shared<TYPE>(value_field_->WithType(value_builder_->type()));
136  }
137 
138  protected:
139  TypedBufferBuilder<offset_type> offsets_builder_;
140  std::shared_ptr<ArrayBuilder> value_builder_;
141  std::shared_ptr<Field> value_field_;
142 
143  Status CheckNextOffset() const {
144  const int64_t num_values = value_builder_->length();
145  ARROW_RETURN_IF(
146  num_values > maximum_elements(),
147  Status::CapacityError("List array cannot contain more than ", maximum_elements(),
148  " child elements,", " have ", num_values));
149  return Status::OK();
150  }
151 
152  Status AppendNextOffset() {
153  ARROW_RETURN_NOT_OK(CheckNextOffset());
154  const int64_t num_values = value_builder_->length();
155  return offsets_builder_.Append(static_cast<offset_type>(num_values));
156  }
157 };
