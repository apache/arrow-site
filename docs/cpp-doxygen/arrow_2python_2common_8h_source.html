49 // Catch a pending Python exception and return the corresponding Status.
50 // If no exception is pending, Status::OK() is returned.
51 inline Status CheckPyError(StatusCode code = StatusCode::UnknownError) {
52  if (ARROW_PREDICT_TRUE(!PyErr_Occurred())) {
53  return Status::OK();
54  } else {
55  return ConvertPyError(code);
56  }
57 }

63 // For Cython, as you can't define template C++ functions in Cython, only use them.
64 // This function can set a Python exception. It assumes that T has a (cheap)
65 // default constructor.
66 template <class T>
67 T GetResultValue(Result<T> result) {
68  if (ARROW_PREDICT_TRUE(result.ok())) {
69  return *std::move(result);
70  } else {
71  int r = internal::check_status(result.status()); // takes the GIL
72  assert(r == -1); // should have errored out
73  ARROW_UNUSED(r);
74  return {};
75  }
76 }

78 // A RAII-style helper that ensures the GIL is acquired inside a lexical block.
79 class ARROW_PYTHON_EXPORT PyAcquireGIL {
80  public:
81  PyAcquireGIL() : acquired_gil_(false) { acquire(); }
82 
83  ~PyAcquireGIL() { release(); }
84 
85  void acquire() {
86  if (!acquired_gil_) {
87  state_ = PyGILState_Ensure();
88  acquired_gil_ = true;
89  }
90  }
91 
92  // idempotent
93  void release() {
94  if (acquired_gil_) {
95  PyGILState_Release(state_);
96  acquired_gil_ = false;
97  }
98  }
99 
100  private:
101  bool acquired_gil_;
102  PyGILState_STATE state_;
103  ARROW_DISALLOW_COPY_AND_ASSIGN(PyAcquireGIL);
104 };

118 // A helper to call safely into the Python interpreter from arbitrary C++ code.
119 // The GIL is acquired, and the current thread's error status is preserved.
120 template <typename Function>
121 auto SafeCallIntoPython(Function&& func) -> decltype(func()) {
122  PyAcquireGIL lock;
123  PyObject* exc_type;
124  PyObject* exc_value;
125  PyObject* exc_traceback;
126  PyErr_Fetch(&exc_type, &exc_value, &exc_traceback);
127  auto maybe_status = std::forward<Function>(func)();
128  // If the return Status is a "Python error", the current Python error status
129  // describes the error and shouldn't be clobbered.
130  if (!IsPyError(::arrow::internal::GenericToStatus(maybe_status)) &&
131  exc_type != NULLPTR) {
132  PyErr_Restore(exc_type, exc_value, exc_traceback);
133  }
134  return maybe_status;
135 }

139 // A RAII primitive that DECREFs the underlying PyObject* when it
140 // goes out of scope.
141 class ARROW_PYTHON_EXPORT OwnedRef {
142  public:
143  OwnedRef() : obj_(NULLPTR) {}
144  OwnedRef(OwnedRef&& other) : OwnedRef(other.detach()) {}
145  explicit OwnedRef(PyObject* obj) : obj_(obj) {}
146 
147  OwnedRef& operator=(OwnedRef&& other) {
148  obj_ = other.detach();
149  return *this;
150  }
151 
152  ~OwnedRef() { reset(); }
153 
154  void reset(PyObject* obj) {
155  Py_XDECREF(obj_);
156  obj_ = obj;
157  }
158 
159  void reset() { reset(NULLPTR); }
160 
161  PyObject* detach() {
162  PyObject* result = obj_;
163  obj_ = NULLPTR;
164  return result;
165  }
166 
167  PyObject* obj() const { return obj_; }
168 
169  PyObject** ref() { return &obj_; }
170 
171  operator bool() const { return obj_ != NULLPTR; }
172 
173  private:
174  ARROW_DISALLOW_COPY_AND_ASSIGN(OwnedRef);
175 
176  PyObject* obj_;

213  Status FromUnicode(PyObject* obj) {
214 #if PY_MAJOR_VERSION >= 3
215  Py_ssize_t size;
216  // The utf-8 representation is cached on the unicode object
217  const char* data = PyUnicode_AsUTF8AndSize(obj, &size);
218  RETURN_IF_PYERROR();
219  this->bytes = data;
220  this->size = size;
221  this->ref.reset();
222 #else
223  PyObject* converted = PyUnicode_AsUTF8String(obj);
224  RETURN_IF_PYERROR();
225  this->bytes = PyBytes_AS_STRING(converted);
226  this->size = PyBytes_GET_SIZE(converted);
227  this->ref.reset(converted);
228 #endif
229  return Status::OK();
230  }
231 
232  protected:
233  PyBytesView(const char* b, Py_ssize_t s, PyObject* obj = NULLPTR)
234  : bytes(b), size(s), ref(obj) {}
235 
236  // View the given Python object as string-like, i.e. str or (utf8) bytes
237  Status FromString(PyObject* obj, bool check_utf8, bool* is_utf8) {
238  if (PyUnicode_Check(obj)) {
239  *is_utf8 = true;
240  return FromUnicode(obj);
241  } else {
242  ARROW_RETURN_NOT_OK(FromBinary(obj, "a string or bytes object"));
243  if (check_utf8) {
244  // Check the bytes are utf8 utf-8
245  OwnedRef decoded(PyUnicode_FromStringAndSize(bytes, size));
246  if (ARROW_PREDICT_TRUE(!PyErr_Occurred())) {
247  *is_utf8 = true;
248  } else {
249  *is_utf8 = false;
250  PyErr_Clear();
251  }
252  } else {
253  *is_utf8 = false;
254  }
255  return Status::OK();
256  }
257  }

259  Status FromBinary(PyObject* obj, const char* expected_msg) {
260  if (PyBytes_Check(obj)) {
