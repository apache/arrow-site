38  void Visit(const VectorReadValidityDex& dex) {
39  int idx = dex.ValidityIdx();
40  auto bitmap = eval_batch_.GetBuffer(idx);
41  // The bitmap could be null. Ignore it in this case.
42  if (bitmap != NULLPTR) {
43  src_maps_.push_back(bitmap);
44  src_map_offsets_.push_back(eval_batch_.GetBufferOffset(idx));
45  }
46  }

51  src_maps_.push_back(bitmap);
52  src_map_offsets_.push_back(0); // local bitmap has offset 0
53  }
