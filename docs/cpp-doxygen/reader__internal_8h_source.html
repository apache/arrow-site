61 // Abstraction to decouple row group iteration details from the ColumnReader,
62 // so we can read only a single row group if we want
63 class FileColumnIterator {
64  public:
65  explicit FileColumnIterator(int column_index, ParquetFileReader* reader,
66  std::vector<int> row_groups)
67  : column_index_(column_index),
68  reader_(reader),
69  schema_(reader->metadata()->schema()),
70  row_groups_(row_groups.begin(), row_groups.end()) {}
71 
72  virtual ~FileColumnIterator() {}
73 
74  std::unique_ptr<::parquet::PageReader> NextChunk() {
75  if (row_groups_.empty()) {
76  return nullptr;
77  }
78 
79  auto row_group_reader = reader_->RowGroup(row_groups_.front());
80  row_groups_.pop_front();
81  return row_group_reader->GetColumnPageReader(column_index_);
82  }
83 
84  const SchemaDescriptor* schema() const { return schema_; }
85 
86  const ColumnDescriptor* descr() const { return schema_->Column(column_index_); }
87 
88  std::shared_ptr<FileMetaData> metadata() const { return reader_->metadata(); }
89 
90  int column_index() const { return column_index_; }
91 
92  protected:
93  int column_index_;
94  ParquetFileReader* reader_;
95  const SchemaDescriptor* schema_;
96  std::deque<int> row_groups_;
97 };
