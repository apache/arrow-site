63 static inline bool ParseTimestamp(const char* buf, const char* format,
64  bool ignoreTimeInDay, int64_t* out) {
65 #if defined(_MSC_VER)
66  static std::locale lc_all(setlocale(LC_ALL, NULLPTR));
67  std::istringstream stream(buf);
68  stream.imbue(lc_all);
69 
70  // TODO: date::parse fails parsing when the hour value is 0.
71  // eg.1886-12-01 00:00:00
72  arrow_vendored::date::sys_seconds seconds;
73  if (ignoreTimeInDay) {
74  arrow_vendored::date::sys_days days;
75  stream >> arrow_vendored::date::parse(format, days);
76  if (stream.fail()) {
77  return false;
78  }
79  seconds = days;
80  } else {
81  stream >> arrow_vendored::date::parse(format, seconds);
82  if (stream.fail()) {
83  return false;
84  }
85  }
86  auto seconds_in_epoch = seconds.time_since_epoch().count();
87  *out = seconds_in_epoch;
88  return true;
89 #else
90  struct tm result;
91  char* ret = strptime(buf, format, &result);
92  if (ret == NULLPTR) {
93  return false;
94  }
95  // ignore the time part
96  arrow_vendored::date::sys_seconds secs =
97  arrow_vendored::date::sys_days(arrow_vendored::date::year(result.tm_year + 1900) /
98  (result.tm_mon + 1) / result.tm_mday);
99  if (!ignoreTimeInDay) {
100  secs += (std::chrono::hours(result.tm_hour) + std::chrono::minutes(result.tm_min) +
101  std::chrono::seconds(result.tm_sec));
102  }
103  *out = secs.time_since_epoch().count();
104  return true;
105 #endif
106 }
