27 // modified from boost LRU cache -> the boost cache supported only an
28 // ordered map.
29 namespace gandiva {
30 // a cache which evicts the least recently used item when it is full
31 template <class Key, class Value>
32 class LruCache {
33  public:
34  using key_type = Key;
35  using value_type = Value;
36  using list_type = std::list<key_type>;
37  struct hasher {
38  template <typename I>
39  std::size_t operator()(const I& i) const {
40  return i.Hash();
41  }
42  };
43  using map_type =
44  std::unordered_map<key_type, std::pair<value_type, typename list_type::iterator>,
45  hasher>;
46 
47  explicit LruCache(size_t capacity) : cache_capacity_(capacity) {}
48 
49  ~LruCache() {}
50 
51  size_t size() const { return map_.size(); }
52 
53  size_t capacity() const { return cache_capacity_; }
54 
55  bool empty() const { return map_.empty(); }
56 
57  bool contains(const key_type& key) { return map_.find(key) != map_.end(); }
58 
59  void insert(const key_type& key, const value_type& value) {
60  typename map_type::iterator i = map_.find(key);
61  if (i == map_.end()) {
62  // insert item into the cache, but first check if it is full
63  if (size() >= cache_capacity_) {
64  // cache is full, evict the least recently used item
65  evict();
66  }
67 
68  // insert the new item
69  lru_list_.push_front(key);
70  map_[key] = std::make_pair(value, lru_list_.begin());
71  }
72  }
73 
74  boost::optional<value_type> get(const key_type& key) {
75  // lookup value in the cache
76  typename map_type::iterator value_for_key = map_.find(key);
77  if (value_for_key == map_.end()) {
78  // value not in cache
79  return boost::none;
80  }
81 
82  // return the value, but first update its place in the most
83  // recently used list
84  typename list_type::iterator position_in_lru_list = value_for_key->second.second;
85  if (position_in_lru_list != lru_list_.begin()) {
86  // move item to the front of the most recently used list
87  lru_list_.erase(position_in_lru_list);
88  lru_list_.push_front(key);
89 
90  // update iterator in map
91  position_in_lru_list = lru_list_.begin();
92  const value_type& value = value_for_key->second.first;
93  map_[key] = std::make_pair(value, position_in_lru_list);
94 
95  // return the value
96  return value;
97  } else {
98  // the item is already at the front of the most recently
99  // used list so just return it
100  return value_for_key->second.first;
101  }
102  }
103 
104  void clear() {
105  map_.clear();
106  lru_list_.clear();
107  }
108 
109  private:
110  void evict() {
111  // evict item from the end of most recently used list
112  typename list_type::iterator i = --lru_list_.end();
113  map_.erase(*i);
114  lru_list_.erase(i);
115  }
116 
117  private:
118  map_type map_;
119  list_type lru_list_;
120  size_t cache_capacity_;
121 };
122 } // namespace gandiva
