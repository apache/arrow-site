79 template <typename VISITOR>
80 inline Status VisitTypeInline(const DataType& type, VISITOR* visitor) {
81  switch (type.id()) {
82  ARROW_GENERATE_FOR_ALL_TYPES(TYPE_VISIT_INLINE);
83  case Type::INTERVAL: {
84  const auto& interval_type = dynamic_cast<const IntervalType&>(type);
85  if (interval_type.interval_type() == IntervalType::MONTHS) {
86  return visitor->Visit(internal::checked_cast<const MonthIntervalType&>(type));
87  }
88  if (interval_type.interval_type() == IntervalType::DAY_TIME) {
89  return visitor->Visit(internal::checked_cast<const DayTimeIntervalType&>(type));
90  }
91  break;
92  }
93  default:
94  break;
95  }
96  return Status::NotImplemented("Type not implemented");
97 }

128 // Visit an array's data values, in order, without overhead.
129 //
130 // The Visit function's `visitor` argument should define two public methods:
131 // - Status VisitNull()
132 // - Status VisitValue(<scalar>)
133 //
134 // The scalar value's type depends on the array data type:
135 // - the type's `c_type`, if any
136 // - for boolean arrays, a `bool`
137 // - for binary, string and fixed-size binary arrays, a `util::string_view`

142 template <>
143 struct ArrayDataVisitor<BooleanType> {
144  template <typename Visitor>
145  static Status Visit(const ArrayData& arr, Visitor* visitor) {
146  if (arr.null_count != 0) {
147  internal::BitmapReader valid_reader(arr.buffers[0]->data(), arr.offset, arr.length);
148  internal::BitmapReader value_reader(arr.buffers[1]->data(), arr.offset, arr.length);
149  for (int64_t i = 0; i < arr.length; ++i) {
150  const bool is_null = valid_reader.IsNotSet();
151  if (is_null) {
152  ARROW_RETURN_NOT_OK(visitor->VisitNull());
153  } else {
154  ARROW_RETURN_NOT_OK(visitor->VisitValue(value_reader.IsSet()));
155  }
156  valid_reader.Next();
157  value_reader.Next();
158  }
159  } else {
160  internal::BitmapReader value_reader(arr.buffers[1]->data(), arr.offset, arr.length);
161  for (int64_t i = 0; i < arr.length; ++i) {
162  ARROW_RETURN_NOT_OK(visitor->VisitValue(value_reader.IsSet()));
163  value_reader.Next();
164  }
165  }
166  return Status::OK();
167  }
168 };

172  template <typename Visitor>
173  static Status Visit(const ArrayData& arr, Visitor* visitor) {
174  using c_type = typename T::c_type;
175  const c_type* data = arr.GetValues<c_type>(1);

177  if (arr.null_count != 0) {
178  internal::BitmapReader valid_reader(arr.buffers[0]->data(), arr.offset, arr.length);
179  for (int64_t i = 0; i < arr.length; ++i) {
180  const bool is_null = valid_reader.IsNotSet();
181  if (is_null) {
182  ARROW_RETURN_NOT_OK(visitor->VisitNull());
183  } else {
184  ARROW_RETURN_NOT_OK(visitor->VisitValue(data[i]));

186  valid_reader.Next();

197 template <typename T>
198 struct ArrayDataVisitor<T, enable_if_base_binary<T>> {
199  template <typename Visitor>
200  static Status Visit(const ArrayData& arr, Visitor* visitor) {
201  using offset_type = typename T::offset_type;
202  constexpr uint8_t empty_value = 0;
203 
204  const offset_type* offsets = arr.GetValues<offset_type>(1);
205  const uint8_t* data;
206  if (!arr.buffers[2]) {
207  data = &empty_value;
208  } else {
209  // Do not use array offset here, as the sliced offsets array refers
210  // to the non-sliced values array.
211  data = arr.GetValues<uint8_t>(2, /*absolute_offset=*/0);
212  }
213 
214  if (arr.null_count != 0) {
215  internal::BitmapReader valid_reader(arr.buffers[0]->data(), arr.offset, arr.length);
216  for (int64_t i = 0; i < arr.length; ++i) {
217  const bool is_null = valid_reader.IsNotSet();
218  valid_reader.Next();
219  if (is_null) {
220  ARROW_RETURN_NOT_OK(visitor->VisitNull());
221  } else {
222  auto value = util::string_view(reinterpret_cast<const char*>(data + offsets[i]),
223  offsets[i + 1] - offsets[i]);
224  ARROW_RETURN_NOT_OK(visitor->VisitValue(value));
225  }
226  }
227  } else {
228  for (int64_t i = 0; i < arr.length; ++i) {
229  auto value = util::string_view(reinterpret_cast<const char*>(data + offsets[i]),
230  offsets[i + 1] - offsets[i]);
231  ARROW_RETURN_NOT_OK(visitor->VisitValue(value));
232  }
233  }
234  return Status::OK();
235  }

238 template <typename T>
239 struct ArrayDataVisitor<T, enable_if_fixed_size_binary<T>> {
240  template <typename Visitor>
241  static Status Visit(const ArrayData& arr, Visitor* visitor) {
242  const auto& fw_type = internal::checked_cast<const FixedSizeBinaryType&>(*arr.type);
243 
244  const int32_t byte_width = fw_type.byte_width();
245  const uint8_t* data =
246  arr.GetValues<uint8_t>(1,
247  /*absolute_offset=*/arr.offset * byte_width);
248 
249  if (arr.null_count != 0) {
250  internal::BitmapReader valid_reader(arr.buffers[0]->data(), arr.offset, arr.length);
251  for (int64_t i = 0; i < arr.length; ++i) {
252  const bool is_null = valid_reader.IsNotSet();
253  valid_reader.Next();
254  if (is_null) {
255  ARROW_RETURN_NOT_OK(visitor->VisitNull());
256  } else {
257  auto value = util::string_view(reinterpret_cast<const char*>(data), byte_width);
258  ARROW_RETURN_NOT_OK(visitor->VisitValue(value));
259  }
260  data += byte_width;
261  }
262  } else {
263  for (int64_t i = 0; i < arr.length; ++i) {
264  auto value = util::string_view(reinterpret_cast<const char*>(data), byte_width);
265  ARROW_RETURN_NOT_OK(visitor->VisitValue(value));
266  data += byte_width;
267  }
268  }
269  return Status::OK();
270  }
