376  std::shared_ptr<WriterProperties> build() {
377  std::unordered_map<std::string, ColumnProperties> column_properties;
378  auto get = [&](const std::string& key) -> ColumnProperties& {
379  auto it = column_properties.find(key);
380  if (it == column_properties.end())
381  return column_properties[key] = default_column_properties_;
382  else
383  return it->second;
384  };
385 
386  for (const auto& item : encodings_) get(item.first).set_encoding(item.second);
387  for (const auto& item : codecs_) get(item.first).set_compression(item.second);
388  for (const auto& item : codecs_compression_level_)
389  get(item.first).set_compression_level(item.second);
390  for (const auto& item : dictionary_enabled_)
391  get(item.first).set_dictionary_enabled(item.second);
392  for (const auto& item : statistics_enabled_)
393  get(item.first).set_statistics_enabled(item.second);
394 
395  return std::shared_ptr<WriterProperties>(new WriterProperties(
396  pool_, dictionary_pagesize_limit_, write_batch_size_, max_row_group_length_,
397  pagesize_, version_, created_by_, std::move(file_encryption_properties_),
398  default_column_properties_, column_properties));
399  }
