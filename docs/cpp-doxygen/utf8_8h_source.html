47 // A compact state table allowing UTF8 decoding using two dependent
48 // lookups per byte. The first lookup determines the character class
49 // and the second lookup reads the next state.
50 // In this table states are multiples of 12.
51 ARROW_EXPORT extern const uint8_t utf8_small_table[256 + 9 * 12];
52 
53 // Success / reject states when looked up in the small table
54 static constexpr uint8_t kUTF8DecodeAccept = 0;
55 static constexpr uint8_t kUTF8DecodeReject = 12;
56 
57 // An expanded state table allowing transitions using a single lookup
58 // at the expense of a larger memory footprint (but on non-random data,
59 // not all the table will end up accessed and cached).
60 // In this table states are multiples of 256.
61 ARROW_EXPORT extern uint16_t utf8_large_table[9 * 256];
62 
63 // Success / reject states when looked up in the large table
64 static constexpr uint16_t kUTF8ValidateAccept = 0;
65 static constexpr uint16_t kUTF8ValidateReject = 256;
66 
67 static inline uint8_t DecodeOneUTF8Byte(uint8_t byte, uint8_t state, uint32_t* codep) {
68  uint8_t type = utf8_small_table[byte];
69 
70  *codep = (state != kUTF8DecodeAccept) ? (byte & 0x3fu) | (*codep << 6)
71  : (0xff >> type) & (byte);
72 
73  state = utf8_small_table[256 + state + type];
74  return state;
75 }
76 
77 static inline uint16_t ValidateOneUTF8Byte(uint8_t byte, uint16_t state) {
78  return utf8_large_table[state + byte];
79 }

88 inline bool ValidateUTF8(const uint8_t* data, int64_t size) {
89  static constexpr uint64_t high_bits_64 = 0x8080808080808080ULL;
90  // For some reason, defining this variable outside the loop helps clang
91  uint64_t mask;
92 
93 #ifndef NDEBUG
94  internal::CheckUTF8Initialized();
95 #endif
96 
97  while (size >= 8) {
98  // XXX This is doing an unaligned access. Contemporary architectures
99  // (x86-64, AArch64, PPC64) support it natively and often have good
100  // performance nevertheless.
101  memcpy(&mask, data, 8);
102  if (ARROW_PREDICT_TRUE((mask & high_bits_64) == 0)) {
103  // 8 bytes of pure ASCII, move forward
104  size -= 8;
105  data += 8;
106  continue;
107  }
108  // Non-ASCII run detected.
109  // We process at least 4 bytes, to avoid too many spurious 64-bit reads
110  // in case the non-ASCII bytes are at the end of the tested 64-bit word.
111  // We also only check for rejection at the end since that state is stable
112  // (once in reject state, we always remain in reject state).
113  // It is guaranteed that size >= 8 when arriving here, which allows
114  // us to avoid size checks.
115  uint16_t state = internal::kUTF8ValidateAccept;
116  // Byte 0
117  state = internal::ValidateOneUTF8Byte(*data++, state);
118  --size;
119  // Byte 1
120  state = internal::ValidateOneUTF8Byte(*data++, state);
121  --size;
122  // Byte 2
123  state = internal::ValidateOneUTF8Byte(*data++, state);
124  --size;
125  // Byte 3
126  state = internal::ValidateOneUTF8Byte(*data++, state);
127  --size;
128  // Byte 4
129  state = internal::ValidateOneUTF8Byte(*data++, state);
130  --size;
131  if (state == internal::kUTF8ValidateAccept) {
132  continue; // Got full char, switch back to ASCII detection
133  }
134  // Byte 5
135  state = internal::ValidateOneUTF8Byte(*data++, state);
136  --size;
137  if (state == internal::kUTF8ValidateAccept) {
138  continue; // Got full char, switch back to ASCII detection
139  }
140  // Byte 6
141  state = internal::ValidateOneUTF8Byte(*data++, state);
142  --size;
143  if (state == internal::kUTF8ValidateAccept) {
144  continue; // Got full char, switch back to ASCII detection
145  }
146  // Byte 7
147  state = internal::ValidateOneUTF8Byte(*data++, state);
148  --size;
149  if (state == internal::kUTF8ValidateAccept) {
150  continue; // Got full char, switch back to ASCII detection
151  }
152  // kUTF8ValidateAccept not reached along 4 transitions has to mean a rejection
153  assert(state == internal::kUTF8ValidateReject);
154  return false;
155  }
156 
157  // Validate string tail one byte at a time
158  // Note the state table is designed so that, once in the reject state,
159  // we remain in that state until the end. So we needn't check for
160  // rejection at each char (we don't gain much by short-circuiting here).
161  uint16_t state = internal::kUTF8ValidateAccept;
162  while (size-- > 0) {
163  state = internal::ValidateOneUTF8Byte(*data++, state);
164  }
165  return ARROW_PREDICT_TRUE(state == internal::kUTF8ValidateAccept);
166 }

169  const uint8_t* data = reinterpret_cast<const uint8_t*>(str.data());
