102 struct FileSystemFactoryOptions {
103  // Either an explicit Partitioning or a PartitioningFactory to discover one.
104  //
105  // If a factory is provided, it will be used to infer a schema for partition fields
106  // based on file and directory paths then construct a Partitioning. The default
107  // is a Partitioning which will yield no partition information.
108  //
109  // The (explicit or discovered) partitioning will be applied to discovered files
110  // and the resulting partition information embedded in the Source.
111  PartitioningOrFactory partitioning{Partitioning::Default()};
112 
113  // For the purposes of applying the partitioning, paths will be stripped
114  // of the partition_base_dir. Files not matching the partition_base_dir
115  // prefix will be skipped for partition discovery. The ignored files will still
116  // be part of the Source, but will not have partition information.
117  //
118  // Example:
119  // partition_base_dir = "/dataset";
120  //
121  // - "/dataset/US/sales.csv" -> "US/sales.csv" will be given to the partitioning
122  //
123  // - "/home/john/late_sales.csv" -> Will be ignored for partition discovery.
124  //
125  // This is useful for partitioning which parses directory when ordering
126  // is important, e.g. DirectoryPartitioning.
127  std::string partition_base_dir;
128 
129  // Invalid files (via selector or explicitly) will be excluded by checking
130  // with the FileFormat::IsSupported method. This will incur IO for each files
131  // in a serial and single threaded fashion. Disabling this feature will skip the
132  // IO, but unsupported files may be present in the Source
133  // (resulting in an error at scan time).
134  bool exclude_invalid_files = true;
135 
136  // Files matching one of the following prefix will be ignored by the
137  // discovery process. This is matched to the basename of a path.
138  //
139  // Example:
140  // ignore_prefixes = {"_", ".DS_STORE" };
141  //
142  // - "/dataset/data.csv" -> not ignored
143  // - "/dataset/_metadata" -> ignored
144  // - "/dataset/.DS_STORE" -> ignored
145  std::vector<std::string> ignore_prefixes = {
146  ".",
147  "_",
148  };
149 };
