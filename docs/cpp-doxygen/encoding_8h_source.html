266 template <typename DType>
267 class TypedDecoder : virtual public Decoder {
268  public:
269  using T = typename DType::c_type;
270 
279  virtual int Decode(T* buffer, int max_values) = 0;
280 
290  virtual int DecodeSpaced(T* buffer, int num_values, int null_count,
291  const uint8_t* valid_bits, int64_t valid_bits_offset) {
292  int values_to_read = num_values - null_count;
293  int values_read = Decode(buffer, values_to_read);
294  if (values_read != values_to_read) {
295  throw ParquetException("Number of values / definition_levels read did not match");
296  }
297 
298  // Depending on the number of nulls, some of the value slots in buffer may
299  // be uninitialized, and this will cause valgrind warnings / potentially UB
300  memset(static_cast<void*>(buffer + values_read), 0,
301  (num_values - values_read) * sizeof(T));
302 
303  // Add spacing for null entries. As we have filled the buffer from the front,
304  // we need to add the spacing from the back.
305  int values_to_move = values_read;
306  for (int i = num_values - 1; i >= 0; i--) {
307  if (BitUtil::GetBit(valid_bits, valid_bits_offset + i)) {
308  buffer[i] = buffer[--values_to_move];
309  }
310  }
311  return num_values;
312  }
313 
317  virtual int DecodeArrow(int num_values, int null_count, const uint8_t* valid_bits,
318  int64_t valid_bits_offset,
319  typename EncodingTraits<DType>::Accumulator* out) = 0;
320 
324  virtual int DecodeArrowNonNull(int num_values,
325  typename EncodingTraits<DType>::Accumulator* out) {
