37 class MemoryPoolStats {

41  int64_t max_memory() const { return max_memory_.load(); }
42 
43  int64_t bytes_allocated() const { return bytes_allocated_.load(); }
44 
45  inline void UpdateAllocatedBytes(int64_t diff) {
46  auto allocated = bytes_allocated_.fetch_add(diff) + diff;
47  // "maximum" allocated memory is ill-defined in multi-threaded code,
48  // so don't try to be too rigorous here
49  if (diff > 0 && allocated > max_memory_) {
50  max_memory_ = allocated;
51  }
52  }
53 
54  protected:
55  std::atomic<int64_t> bytes_allocated_;
56  std::atomic<int64_t> max_memory_;
57 };
