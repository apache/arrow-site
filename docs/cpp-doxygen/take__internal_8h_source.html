67 template <bool SomeIndicesNull, bool SomeValuesNull, bool NeverOutOfBounds,
68  typename IndexSequence, typename Visitor>
69 Status VisitIndices(IndexSequence indices, const Array& values, Visitor&& vis) {
70  for (int64_t i = 0; i < indices.length(); ++i) {
71  auto index_valid = indices.Next();
72  if (SomeIndicesNull && !index_valid.second) {
73  RETURN_NOT_OK(vis(0, false));
74  continue;
75  }
76 
77  auto index = index_valid.first;
78  if (!NeverOutOfBounds) {
79  if (index < 0 || index >= values.length()) {
80  return Status::IndexError("take index out of bounds");
81  }
82  } else {
83  DCHECK_GE(index, 0);
84  DCHECK_LT(index, values.length());
85  }
86 
87  bool is_valid = !SomeValuesNull || values.IsValid(index);
88  RETURN_NOT_OK(vis(index, is_valid));
89  }
90  return Status::OK();
91 }

93 template <bool SomeIndicesNull, bool SomeValuesNull, typename IndexSequence,
94  typename Visitor>
95 Status VisitIndices(IndexSequence indices, const Array& values, Visitor&& vis) {
96  if (indices.never_out_of_bounds()) {
97  return VisitIndices<SomeIndicesNull, SomeValuesNull, true>(
98  indices, values, std::forward<Visitor>(vis));
99  }
100  return VisitIndices<SomeIndicesNull, SomeValuesNull, false>(indices, values,
101  std::forward<Visitor>(vis));
102 }

104 template <bool SomeIndicesNull, typename IndexSequence, typename Visitor>
105 Status VisitIndices(IndexSequence indices, const Array& values, Visitor&& vis) {
106  if (values.null_count() == 0) {
107  return VisitIndices<SomeIndicesNull, false>(indices, values,
108  std::forward<Visitor>(vis));
109  }
110  return VisitIndices<SomeIndicesNull, true>(indices, values, std::forward<Visitor>(vis));
111 }

113 template <typename IndexSequence, typename Visitor>
114 Status VisitIndices(IndexSequence indices, const Array& values, Visitor&& vis) {
115  if (indices.null_count() == 0) {
116  return VisitIndices<false>(indices, values, std::forward<Visitor>(vis));
117  }
118  return VisitIndices<true>(indices, values, std::forward<Visitor>(vis));

121 // Helper class for gathering values from an array
122 template <typename IndexSequence>
123 class Taker {
124  public:
125  explicit Taker(const std::shared_ptr<DataType>& type) : type_(type) {}
126 
127  virtual ~Taker() = default;
128 
129  // initialize this taker including constructing any children,
130  // must be called once after construction before any other methods are called
131  virtual Status Init() { return Status::OK(); }
132 
133  // reset this Taker and set FunctionContext for taking an array
134  // must be called each time the FunctionContext may have changed
135  virtual Status SetContext(FunctionContext* ctx) = 0;
136 
137  // gather elements from an array at the provided indices
138  virtual Status Take(const Array& values, IndexSequence indices) = 0;
139 
140  // assemble an array of all gathered values
141  virtual Status Finish(std::shared_ptr<Array>*) = 0;
142 
143  // factory; the output Taker will support gathering values of the given type
144  static Status Make(const std::shared_ptr<DataType>& type, std::unique_ptr<Taker>* out);
145 
146  static_assert(std::is_literal_type<IndexSequence>::value,
147  "Index sequences must be literal type");
148 
149  static_assert(std::is_copy_constructible<IndexSequence>::value,
150  "Index sequences must be copy constructible");
151 
152  static_assert(std::is_same<decltype(std::declval<IndexSequence>().Next()),
153  std::pair<int64_t, bool>>::value,
154  "An index sequence must yield pairs of indices:int64_t, validity:bool.");
155 
156  static_assert(std::is_same<decltype(std::declval<const IndexSequence>().length()),
157  int64_t>::value,
158  "An index sequence must provide its length.");
159 
160  static_assert(std::is_same<decltype(std::declval<const IndexSequence>().null_count()),
161  int64_t>::value,
162  "An index sequence must provide the number of nulls it will take.");
163 
164  static_assert(
165  std::is_same<decltype(std::declval<const IndexSequence>().never_out_of_bounds()),
166  bool>::value,
167  "Index sequences must declare whether bounds checking is necessary");
168 
169  static_assert(
170  std::is_same<decltype(std::declval<IndexSequence>().set_never_out_of_bounds()),
171  void>::value,
172  "An index sequence must support ignoring bounds checking.");
173 
174  protected:
175  template <typename Builder>
176  Status MakeBuilder(MemoryPool* pool, std::unique_ptr<Builder>* out) {
177  std::unique_ptr<ArrayBuilder> builder;
178  RETURN_NOT_OK(arrow::MakeBuilder(pool, type_, &builder));
179  out->reset(checked_cast<Builder*>(builder.release()));
180  return Status::OK();
181  }
182 
183  std::shared_ptr<DataType> type_;
184 };

187 // or yields null for the length of that range
188 class RangeIndexSequence {
189  public:
190  constexpr bool never_out_of_bounds() const { return true; }
191  void set_never_out_of_bounds() {}

196  : is_valid_(is_valid), index_(offset), length_(length) {}

198  std::pair<int64_t, bool> Next() { return std::make_pair(index_++, is_valid_); }

201 
202  int64_t null_count() const { return is_valid_ ? 0 : length_; }

209 // an IndexSequence which yields the values of an Array of integers
210 template <typename IndexType>
211 class ArrayIndexSequence {
212  public:
213  bool never_out_of_bounds() const { return never_out_of_bounds_; }
214  void set_never_out_of_bounds() { never_out_of_bounds_ = true; }
215 
216  constexpr ArrayIndexSequence() = default;
217 
218  explicit ArrayIndexSequence(const Array& indices)
219  : indices_(&checked_cast<const NumericArray<IndexType>&>(indices)) {}
220 
221  std::pair<int64_t, bool> Next() {
222  if (indices_->IsNull(index_)) {
223  ++index_;
224  return std::make_pair(-1, false);
225  }
226  return std::make_pair(indices_->Value(index_++), true);
227  }
228 
229  int64_t length() const { return indices_->length(); }
230 
231  int64_t null_count() const { return indices_->null_count(); }
232 
233  private:
234  const NumericArray<IndexType>* indices_ = nullptr;
235  int64_t index_ = 0;
236  bool never_out_of_bounds_ = false;
237 };

239 // Default implementation: taking from a simple array into a builder requires only that
240 // the array supports array.GetView() and the corresponding builder supports
241 // builder.UnsafeAppend(array.GetView())
242 template <typename IndexSequence, typename T>
243 class TakerImpl : public Taker<IndexSequence> {
244  public:
245  using ArrayType = typename TypeTraits<T>::ArrayType;
246  using BuilderType = typename TypeTraits<T>::BuilderType;
247 
248  using Taker<IndexSequence>::Taker;
249 
250  Status SetContext(FunctionContext* ctx) override {
251  return this->MakeBuilder(ctx->memory_pool(), &builder_);
252  }
253 
254  Status Take(const Array& values, IndexSequence indices) override {
255  DCHECK(this->type_->Equals(values.type()));
256  RETURN_NOT_OK(builder_->Reserve(indices.length()));
257  return VisitIndices(indices, values, [&](int64_t index, bool is_valid) {
258  if (!is_valid) {
259  builder_->UnsafeAppendNull();
260  return Status::OK();
261  }
262  auto value = checked_cast<const ArrayType&>(values).GetView(index);
263  return UnsafeAppend(builder_.get(), value);
264  });
265  }
266 
267  Status Finish(std::shared_ptr<Array>* out) override { return builder_->Finish(out); }
268 
269  private:
270  std::unique_ptr<BuilderType> builder_;
271 };

303 template <typename IndexSequence, typename TypeClass>
304 class ListTakerImpl : public Taker<IndexSequence> {
305  public:
306  using offset_type = typename TypeClass::offset_type;
307  using ArrayType = typename TypeTraits<TypeClass>::ArrayType;
308 
309  using Taker<IndexSequence>::Taker;
310 
311  Status Init() override {
312  const auto& list_type = checked_cast<const TypeClass&>(*this->type_);
313  return Taker<RangeIndexSequence>::Make(list_type.value_type(), &value_taker_);
314  }
315 
316  Status SetContext(FunctionContext* ctx) override {
317  auto pool = ctx->memory_pool();
318  null_bitmap_builder_.reset(new TypedBufferBuilder<bool>(pool));
319  offset_builder_.reset(new TypedBufferBuilder<offset_type>(pool));
320  RETURN_NOT_OK(offset_builder_->Append(0));
321  return value_taker_->SetContext(ctx);
322  }
323 
324  Status Take(const Array& values, IndexSequence indices) override {
325  DCHECK(this->type_->Equals(values.type()));
326 
327  const auto& list_array = checked_cast<const ArrayType&>(values);
328 
329  RETURN_NOT_OK(null_bitmap_builder_->Reserve(indices.length()));
330  RETURN_NOT_OK(offset_builder_->Reserve(indices.length()));
331 
332  offset_type offset = offset_builder_->data()[offset_builder_->length() - 1];
333  return VisitIndices(indices, values, [&](int64_t index, bool is_valid) {
334  null_bitmap_builder_->UnsafeAppend(is_valid);
335 
336  if (is_valid) {
337  offset += list_array.value_length(index);
338  RangeIndexSequence value_indices(true, list_array.value_offset(index),
339  list_array.value_length(index));
340  RETURN_NOT_OK(value_taker_->Take(*list_array.values(), value_indices));
341  }
342 
343  offset_builder_->UnsafeAppend(offset);
344  return Status::OK();
345  });
346  }
347 
348  Status Finish(std::shared_ptr<Array>* out) override {
349  auto null_count = null_bitmap_builder_->false_count();
350  auto length = null_bitmap_builder_->length();
351 
352  std::shared_ptr<Buffer> offsets, null_bitmap;
353  RETURN_NOT_OK(null_bitmap_builder_->Finish(&null_bitmap));
354  RETURN_NOT_OK(offset_builder_->Finish(&offsets));
355 
356  std::shared_ptr<Array> taken_values;
357  RETURN_NOT_OK(value_taker_->Finish(&taken_values));
358 
359  out->reset(new ArrayType(this->type_, length, offsets, taken_values, null_bitmap,
360  null_count));
361  return Status::OK();
362  }
363 
364  std::unique_ptr<TypedBufferBuilder<bool>> null_bitmap_builder_;
365  std::unique_ptr<TypedBufferBuilder<offset_type>> offset_builder_;
366  std::unique_ptr<Taker<RangeIndexSequence>> value_taker_;
367 };

441  public:
442  using Taker<IndexSequence>::Taker;
443 
444  Status Init() override {
445  children_.resize(this->type_->num_children());
446  for (int i = 0; i < this->type_->num_children(); ++i) {
447  RETURN_NOT_OK(
448  Taker<IndexSequence>::Make(this->type_->child(i)->type(), &children_[i]));

452

454  null_bitmap_builder_.reset(new TypedBufferBuilder<bool>(ctx->memory_pool()));

456  RETURN_NOT_OK(children_[i]->SetContext(ctx));
457  }
458  return Status::OK();
459  }
460 
461  Status Take(const Array& values, IndexSequence indices) override {
462  DCHECK(this->type_->Equals(values.type()));
463 
464  RETURN_NOT_OK(null_bitmap_builder_->Reserve(indices.length()));
465  RETURN_NOT_OK(VisitIndices(indices, values, [&](int64_t, bool is_valid) {
466  null_bitmap_builder_->UnsafeAppend(is_valid);
467  return Status::OK();
468  }));
469 
470  // bounds checking was done while appending to the null bitmap
471  indices.set_never_out_of_bounds();
472 
473  const auto& struct_array = checked_cast<const StructArray&>(values);
474  for (int i = 0; i < this->type_->num_children(); ++i) {

476  }
477  return Status::OK();
478  }
479 
480  Status Finish(std::shared_ptr<Array>* out) override {
481  auto null_count = null_bitmap_builder_->false_count();
482  auto length = null_bitmap_builder_->length();
483  std::shared_ptr<Buffer> null_bitmap;
484  RETURN_NOT_OK(null_bitmap_builder_->Finish(&null_bitmap));
485 
486  ArrayVector fields(this->type_->num_children());
487  for (int i = 0; i < this->type_->num_children(); ++i) {
488  RETURN_NOT_OK(children_[i]->Finish(&fields[i]));
489  }
490 
491  out->reset(

493  return Status::OK();
494  }

496  protected:
497  std::unique_ptr<TypedBufferBuilder<bool>> null_bitmap_builder_;
498  std::vector<std::unique_ptr<Taker<IndexSequence>>> children_;

501 template <typename IndexSequence>
502 class TakerImpl<IndexSequence, UnionType> : public Taker<IndexSequence> {
503  public:
504  using Taker<IndexSequence>::Taker;
505 
506  Status Init() override {
507  union_type_ = checked_cast<const UnionType*>(this->type_.get());
508 
509  if (union_type_->mode() == UnionMode::SPARSE) {
510  sparse_children_.resize(this->type_->num_children());
511  } else {
512  dense_children_.resize(this->type_->num_children());
513  child_length_.resize(union_type_->max_type_code() + 1);
514  }
515 
516  for (int i = 0; i < this->type_->num_children(); ++i) {
517  if (union_type_->mode() == UnionMode::SPARSE) {
518  RETURN_NOT_OK(Taker<IndexSequence>::Make(this->type_->child(i)->type(),
519  &sparse_children_[i]));
520  } else {
521  RETURN_NOT_OK(Taker<ArrayIndexSequence<Int32Type>>::Make(
522  this->type_->child(i)->type(), &dense_children_[i]));
523  }
524  }
525 
526  return Status::OK();
527  }
528 
529  Status SetContext(FunctionContext* ctx) override {
530  pool_ = ctx->memory_pool();
531  null_bitmap_builder_.reset(new TypedBufferBuilder<bool>(pool_));
532  type_code_builder_.reset(new TypedBufferBuilder<int8_t>(pool_));
533 
534  if (union_type_->mode() == UnionMode::DENSE) {
535  offset_builder_.reset(new TypedBufferBuilder<int32_t>(pool_));
536  std::fill(child_length_.begin(), child_length_.end(), 0);
537  }
538 
539  for (int i = 0; i < this->type_->num_children(); ++i) {
540  if (union_type_->mode() == UnionMode::SPARSE) {
541  RETURN_NOT_OK(sparse_children_[i]->SetContext(ctx));
542  } else {
543  RETURN_NOT_OK(dense_children_[i]->SetContext(ctx));
544  }
545  }
546 
547  return Status::OK();
548  }
549 
550  Status Take(const Array& values, IndexSequence indices) override {
551  DCHECK(this->type_->Equals(values.type()));
552  const auto& union_array = checked_cast<const UnionArray&>(values);
553  auto type_codes = union_array.raw_type_codes();
554 
555  if (union_type_->mode() == UnionMode::SPARSE) {
556  RETURN_NOT_OK(null_bitmap_builder_->Reserve(indices.length()));
557  RETURN_NOT_OK(type_code_builder_->Reserve(indices.length()));
558  RETURN_NOT_OK(VisitIndices(indices, values, [&](int64_t index, bool is_valid) {
559  null_bitmap_builder_->UnsafeAppend(is_valid);
560  type_code_builder_->UnsafeAppend(type_codes[index]);
561  return Status::OK();
562  }));
563 
564  // bounds checking was done while appending to the null bitmap
565  indices.set_never_out_of_bounds();
566 
567  for (int i = 0; i < this->type_->num_children(); ++i) {
568  RETURN_NOT_OK(sparse_children_[i]->Take(*union_array.child(i), indices));
569  }
570  } else {
571  // Gathering from the offsets into child arrays is a bit tricky.
572  std::vector<uint32_t> child_counts(union_type_->max_type_code() + 1);
573  RETURN_NOT_OK(null_bitmap_builder_->Reserve(indices.length()));
574  RETURN_NOT_OK(type_code_builder_->Reserve(indices.length()));
575  RETURN_NOT_OK(VisitIndices(indices, values, [&](int64_t index, bool is_valid) {
576  null_bitmap_builder_->UnsafeAppend(is_valid);
577  type_code_builder_->UnsafeAppend(type_codes[index]);
578  child_counts[type_codes[index]] += is_valid;
579  return Status::OK();
580  }));
581 
582  // bounds checking was done while appending to the null bitmap
583  indices.set_never_out_of_bounds();
584 
585  // Allocate temporary storage for the offsets of all valid slots
586  auto child_offsets_storage_size =
587  std::accumulate(child_counts.begin(), child_counts.end(), 0);
588  std::shared_ptr<Buffer> child_offsets_storage;
589  RETURN_NOT_OK(AllocateBuffer(pool_, child_offsets_storage_size * sizeof(int32_t),
590  &child_offsets_storage));
591 
592  // Partition offsets by type_code: child_offset_partitions[type_code] will
593  // point to storage for child_counts[type_code] offsets
594  std::vector<int32_t*> child_offset_partitions(child_counts.size());
595  auto child_offsets_storage_data = GetInt32(child_offsets_storage);
596  for (auto type_code : union_type_->type_codes()) {
597  child_offset_partitions[type_code] = child_offsets_storage_data;
598  child_offsets_storage_data += child_counts[type_code];
599  }
600  DCHECK_EQ(child_offsets_storage_data - GetInt32(child_offsets_storage),
601  child_offsets_storage_size);
602 
603  // Fill child_offsets_storage with the taken offsets
604  RETURN_NOT_OK(offset_builder_->Reserve(indices.length()));
605  RETURN_NOT_OK(VisitIndices(indices, values, [&](int64_t index, bool is_valid) {
606  auto type_code = type_codes[index];
607  if (is_valid) {
608  offset_builder_->UnsafeAppend(child_length_[type_code]++);
609  *child_offset_partitions[type_code] = union_array.value_offset(index);
610  ++child_offset_partitions[type_code];
611  } else {
612  offset_builder_->UnsafeAppend(0);
613  }
614  return Status::OK();
615  }));
616 
617  // Take from each child at those offsets
618  int64_t taken_offset_begin = 0;
619  for (int i = 0; i < this->type_->num_children(); ++i) {
620  auto type_code = union_type_->type_codes()[i];
621  auto length = child_counts[type_code];
622  Int32Array taken_offsets(length, SliceBuffer(child_offsets_storage,
623  sizeof(int32_t) * taken_offset_begin,
624  sizeof(int32_t) * length));
625  ArrayIndexSequence<Int32Type> child_indices(taken_offsets);
626  child_indices.set_never_out_of_bounds();
627  RETURN_NOT_OK(dense_children_[i]->Take(*union_array.child(i), child_indices));
628  taken_offset_begin += length;
629  }
630  }
631 
632  return Status::OK();
633  }
634 
635  Status Finish(std::shared_ptr<Array>* out) override {
636  auto null_count = null_bitmap_builder_->false_count();
637  auto length = null_bitmap_builder_->length();
638  std::shared_ptr<Buffer> null_bitmap, type_codes;
639  RETURN_NOT_OK(null_bitmap_builder_->Finish(&null_bitmap));
640  RETURN_NOT_OK(type_code_builder_->Finish(&type_codes));
641 
642  std::shared_ptr<Buffer> offsets;
643  if (union_type_->mode() == UnionMode::DENSE) {
644  RETURN_NOT_OK(offset_builder_->Finish(&offsets));
645  }
646 
647  ArrayVector fields(this->type_->num_children());
648  for (int i = 0; i < this->type_->num_children(); ++i) {
649  if (union_type_->mode() == UnionMode::SPARSE) {
650  RETURN_NOT_OK(sparse_children_[i]->Finish(&fields[i]));
651  } else {
652  RETURN_NOT_OK(dense_children_[i]->Finish(&fields[i]));
653  }
654  }
655 
656  out->reset(new UnionArray(this->type_, length, std::move(fields), type_codes, offsets,
657  null_bitmap, null_count));
658  return Status::OK();
659  }
660 
661  protected:
662  int32_t* GetInt32(const std::shared_ptr<Buffer>& b) const {
663  return reinterpret_cast<int32_t*>(b->mutable_data());
664  }

667  MemoryPool* pool_ = nullptr;
668  std::unique_ptr<TypedBufferBuilder<bool>> null_bitmap_builder_;
669  std::unique_ptr<TypedBufferBuilder<int8_t>> type_code_builder_;

671  std::vector<std::unique_ptr<Taker<IndexSequence>>> sparse_children_;
672  std::vector<std::unique_ptr<Taker<ArrayIndexSequence<Int32Type>>>> dense_children_;
673  std::vector<int32_t> child_length_;
