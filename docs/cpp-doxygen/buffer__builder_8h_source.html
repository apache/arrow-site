43 class ARROW_EXPORT BufferBuilder {
44  public:
45  explicit BufferBuilder(MemoryPool* pool ARROW_MEMORY_POOL_DEFAULT)
46  : pool_(pool),
47  data_(/*ensure never null to make ubsan happy and avoid check penalties below*/
48  &util::internal::non_null_filler),
49 
50  capacity_(0),
51  size_(0) {}
52 
60  Status Resize(const int64_t new_capacity, bool shrink_to_fit = true) {
61  // Resize(0) is a no-op
62  if (new_capacity == 0) {
63  return Status::OK();
64  }
65  if (buffer_ == NULLPTR) {
66  ARROW_RETURN_NOT_OK(AllocateResizableBuffer(pool_, new_capacity, &buffer_));
67  } else {
68  ARROW_RETURN_NOT_OK(buffer_->Resize(new_capacity, shrink_to_fit));
69  }
70  capacity_ = buffer_->capacity();
71  data_ = buffer_->mutable_data();
72  return Status::OK();
73  }
74 
80  Status Reserve(const int64_t additional_bytes) {
81  auto min_capacity = size_ + additional_bytes;
82  if (min_capacity <= capacity_) {
83  return Status::OK();
84  }
85  return Resize(GrowByFactor(capacity_, min_capacity), false);
86  }
87 
89  static int64_t GrowByFactor(int64_t current_capacity, int64_t new_capacity) {
90  // Doubling capacity except for large Reserve requests. 2x growth strategy
91  // (versus 1.5x) seems to have slightly better performance when using
92  // jemalloc, but significantly better performance when using the system
93  // allocator. See ARROW-6450 for further discussion
94  return std::max(new_capacity, current_capacity * 2);
95  }
96 
100  Status Append(const void* data, const int64_t length) {
101  if (ARROW_PREDICT_FALSE(size_ + length > capacity_)) {
102  ARROW_RETURN_NOT_OK(Resize(GrowByFactor(capacity_, size_ + length), false));
103  }
104  UnsafeAppend(data, length);
105  return Status::OK();
106  }
107 
111  Status Append(const int64_t num_copies, uint8_t value) {
112  ARROW_RETURN_NOT_OK(Reserve(num_copies));
113  UnsafeAppend(num_copies, value);
114  return Status::OK();
115  }
116 
117  // Advance pointer and zero out memory
118  Status Advance(const int64_t length) { return Append(length, 0); }
119 
120  // Advance pointer, but don't allocate or zero memory
121  void UnsafeAdvance(const int64_t length) { size_ += length; }
122 
123  // Unsafe methods don't check existing size
124  void UnsafeAppend(const void* data, const int64_t length) {
125  memcpy(data_ + size_, data, static_cast<size_t>(length));
126  size_ += length;
127  }
128 
129  void UnsafeAppend(const int64_t num_copies, uint8_t value) {
130  memset(data_ + size_, value, static_cast<size_t>(num_copies));
131  size_ += num_copies;
132  }
133 
143  Status Finish(std::shared_ptr<Buffer>* out, bool shrink_to_fit = true) {
144  ARROW_RETURN_NOT_OK(Resize(size_, shrink_to_fit));
145  if (size_ != 0) buffer_->ZeroPadding();
146  *out = buffer_;
147  if (*out == NULLPTR) {
148  ARROW_RETURN_NOT_OK(AllocateBuffer(pool_, 0, out));
149  }
150  Reset();
151  return Status::OK();
152  }
153 
154  void Reset() {
155  buffer_ = NULLPTR;
156  capacity_ = size_ = 0;
157  }
158 
163  void Rewind(int64_t position) { size_ = position; }
164 
165  int64_t capacity() const { return capacity_; }
166  int64_t length() const { return size_; }
167  const uint8_t* data() const { return data_; }
168  uint8_t* mutable_data() { return data_; }
169 
170  private:
171  std::shared_ptr<ResizableBuffer> buffer_;
172  MemoryPool* pool_;
173  uint8_t* data_;
174  int64_t capacity_;
175  int64_t size_;
176 };

182 template <typename T>
183 class TypedBufferBuilder<
184  T, typename std::enable_if<std::is_arithmetic<T>::value ||
185  std::is_standard_layout<T>::value>::type> {
186  public:
187  explicit TypedBufferBuilder(MemoryPool* pool ARROW_MEMORY_POOL_DEFAULT)
188  : bytes_builder_(pool) {}
189 
190  Status Append(T value) {
191  return bytes_builder_.Append(reinterpret_cast<uint8_t*>(&value), sizeof(T));
192  }
193 
194  Status Append(const T* values, int64_t num_elements) {
195  return bytes_builder_.Append(reinterpret_cast<const uint8_t*>(values),
196  num_elements * sizeof(T));
197  }
198 
199  Status Append(const int64_t num_copies, T value) {
200  ARROW_RETURN_NOT_OK(Reserve(num_copies + length()));
201  UnsafeAppend(num_copies, value);
202  return Status::OK();
203  }
204 
205  void UnsafeAppend(T value) {
206  bytes_builder_.UnsafeAppend(reinterpret_cast<uint8_t*>(&value), sizeof(T));
207  }
208 
209  void UnsafeAppend(const T* values, int64_t num_elements) {
210  bytes_builder_.UnsafeAppend(reinterpret_cast<const uint8_t*>(values),
211  num_elements * sizeof(T));
212  }
213 
214  template <typename Iter>
215  void UnsafeAppend(Iter values_begin, Iter values_end) {
216  int64_t num_elements = static_cast<int64_t>(std::distance(values_begin, values_end));
217  auto data = mutable_data() + length();
218  bytes_builder_.UnsafeAdvance(num_elements * sizeof(T));
219  std::copy(values_begin, values_end, data);
220  }
221 
222  void UnsafeAppend(const int64_t num_copies, T value) {
223  auto data = mutable_data() + length();
224  bytes_builder_.UnsafeAdvance(num_copies * sizeof(T));
225  std::fill(data, data + num_copies, value);
226  }
227 
228  Status Resize(const int64_t new_capacity, bool shrink_to_fit = true) {
229  return bytes_builder_.Resize(new_capacity * sizeof(T), shrink_to_fit);
230  }
231 
232  Status Reserve(const int64_t additional_elements) {
233  return bytes_builder_.Reserve(additional_elements * sizeof(T));
234  }
235 
236  Status Advance(const int64_t length) {
237  return bytes_builder_.Advance(length * sizeof(T));
238  }
239 
240  Status Finish(std::shared_ptr<Buffer>* out, bool shrink_to_fit = true) {
241  return bytes_builder_.Finish(out, shrink_to_fit);
242  }
243 
244  void Reset() { bytes_builder_.Reset(); }
245 
246  int64_t length() const { return bytes_builder_.length() / sizeof(T); }
247  int64_t capacity() const { return bytes_builder_.capacity() / sizeof(T); }
248  const T* data() const { return reinterpret_cast<const T*>(bytes_builder_.data()); }
249  T* mutable_data() { return reinterpret_cast<T*>(bytes_builder_.mutable_data()); }
250 
251  private:
252  BufferBuilder bytes_builder_;
253 };

257 class TypedBufferBuilder<bool> {
258  public:
259  explicit TypedBufferBuilder(MemoryPool* pool ARROW_MEMORY_POOL_DEFAULT)
260  : bytes_builder_(pool) {}
261 
262  Status Append(bool value) {
263  ARROW_RETURN_NOT_OK(Reserve(1));
264  UnsafeAppend(value);
265  return Status::OK();
266  }
267 
268  Status Append(const uint8_t* valid_bytes, int64_t num_elements) {
269  ARROW_RETURN_NOT_OK(Reserve(num_elements));
270  UnsafeAppend(valid_bytes, num_elements);
271  return Status::OK();
272  }
273 
274  Status Append(const int64_t num_copies, bool value) {
275  ARROW_RETURN_NOT_OK(Reserve(num_copies));
276  UnsafeAppend(num_copies, value);
277  return Status::OK();
278  }
279 
280  void UnsafeAppend(bool value) {
281  BitUtil::SetBitTo(mutable_data(), bit_length_, value);
282  if (!value) {
283  ++false_count_;
284  }
285  ++bit_length_;
286  }
287 
288  void UnsafeAppend(const uint8_t* bytes, int64_t num_elements) {
289  if (num_elements == 0) return;
290  int64_t i = 0;
291  internal::GenerateBitsUnrolled(mutable_data(), bit_length_, num_elements, [&] {
292  bool value = bytes[i++];
293  false_count_ += !value;
294  return value;
295  });
296  bit_length_ += num_elements;
297  }
298 
299  void UnsafeAppend(const int64_t num_copies, bool value) {
300  BitUtil::SetBitsTo(mutable_data(), bit_length_, num_copies, value);
301  false_count_ += num_copies * !value;
302  bit_length_ += num_copies;
303  }
304 
305  template <bool count_falses, typename Generator>
306  void UnsafeAppend(const int64_t num_elements, Generator&& gen) {
307  if (num_elements == 0) return;
308 
309  if (count_falses) {
310  internal::GenerateBitsUnrolled(mutable_data(), bit_length_, num_elements, [&] {
311  bool value = gen();
312  false_count_ += !value;
313  return value;
314  });
315  } else {
316  internal::GenerateBitsUnrolled(mutable_data(), bit_length_, num_elements,
317  std::forward<Generator>(gen));
318  }
319  bit_length_ += num_elements;
320  }
321 
322  Status Resize(const int64_t new_capacity, bool shrink_to_fit = true) {
323  const int64_t old_byte_capacity = bytes_builder_.capacity();
324  ARROW_RETURN_NOT_OK(
325  bytes_builder_.Resize(BitUtil::BytesForBits(new_capacity), shrink_to_fit));
326  // Resize() may have chosen a larger capacity (e.g. for padding),
327  // so ask it again before calling memset().
328  const int64_t new_byte_capacity = bytes_builder_.capacity();
329  if (new_byte_capacity > old_byte_capacity) {
330  // The additional buffer space is 0-initialized for convenience,
331  // so that other methods can simply bump the length.
332  memset(mutable_data() + old_byte_capacity, 0,
333  static_cast<size_t>(new_byte_capacity - old_byte_capacity));
334  }
335  return Status::OK();
336  }
337 
338  Status Reserve(const int64_t additional_elements) {
339  return Resize(
340  BufferBuilder::GrowByFactor(bit_length_, bit_length_ + additional_elements),
341  false);
342  }
343 
344  Status Advance(const int64_t length) {
345  ARROW_RETURN_NOT_OK(Reserve(length));
346  bit_length_ += length;
347  false_count_ += length;
348  return Status::OK();
349  }
350 
351  Status Finish(std::shared_ptr<Buffer>* out, bool shrink_to_fit = true) {
352  // set bytes_builder_.size_ == byte size of data
353  bytes_builder_.UnsafeAdvance(BitUtil::BytesForBits(bit_length_) -
354  bytes_builder_.length());
355  bit_length_ = false_count_ = 0;
356  return bytes_builder_.Finish(out, shrink_to_fit);
357  }
358 
359  void Reset() {
360  bytes_builder_.Reset();
361  bit_length_ = false_count_ = 0;
362  }
363 
364  int64_t length() const { return bit_length_; }
365  int64_t capacity() const { return bytes_builder_.capacity() * 8; }
366  const uint8_t* data() const { return bytes_builder_.data(); }
367  uint8_t* mutable_data() { return bytes_builder_.mutable_data(); }
368  int64_t false_count() const { return false_count_; }
369 
370  private:
371  BufferBuilder bytes_builder_;
372  int64_t bit_length_ = 0;
373  int64_t false_count_ = 0;
374 };
