30 static inline RecordBatchIterator FilterRecordBatch(RecordBatchIterator it,
31  const ExpressionEvaluator& evaluator,
32  const Expression& filter,
33  MemoryPool* pool) {
34  return MakeMaybeMapIterator(
35  [&filter, &evaluator, pool](std::shared_ptr<RecordBatch> in) {
36  return evaluator.Evaluate(filter, *in, pool).Map([&](compute::Datum selection) {
37  return evaluator.Filter(selection, in);
38  });
39  },
40  std::move(it));
41 }
