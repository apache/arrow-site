61 inline LocalBitMapsHolder::LocalBitMapsHolder(int num_records, int num_local_bitmaps)
62  : num_records_(num_records) {
63  // alloc an array for the pointers to the bitmaps.
64  if (num_local_bitmaps > 0) {
65  local_bitmaps_array_.reset(new uint8_t*[num_local_bitmaps]);
66  }
67 
68  // 64-bit aligned bitmaps.
69  uint32_t roundUp64Multiple = (num_records_ + 63) >> 6;
70  local_bitmap_size_ = roundUp64Multiple * 8;
71 
72  // Alloc 'num_local_bitmaps_' number of bitmaps, each of capacity 'num_records_'.
73  for (int i = 0; i < num_local_bitmaps; ++i) {
74  // TODO : round-up to a slab friendly multiple.
75  std::unique_ptr<uint8_t> bitmap(new uint8_t[local_bitmap_size_]);
76 
77  // keep pointer to the bitmap in the array.
78  (local_bitmaps_array_.get())[i] = bitmap.get();
79 
80  // pre-fill with 1s (assuming that the probability of is_valid is higher).
81  memset(bitmap.get(), 0xff, local_bitmap_size_);
82  local_bitmaps_vec_.push_back(std::move(bitmap));
83  }
84 }
