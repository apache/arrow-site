52 // Needed because std::packaged_task is not copyable and hence not convertible
53 // to std::function.
54 template <typename R, typename... Args>
55 struct packaged_task_wrapper {
56  using PackagedTask = std::packaged_task<R(Args...)>;
57 
58  explicit packaged_task_wrapper(PackagedTask&& task)
59  : task_(std::make_shared<PackagedTask>(std::forward<PackagedTask>(task))) {}
60 
61  void operator()(Args&&... args) { return (*task_)(std::forward<Args>(args)...); }
62  std::shared_ptr<PackagedTask> task_;
63 };

96  // Spawn a fire-and-forget task on one of the workers.
97  template <typename Function>
98  Status Spawn(Function&& func) {
99  return SpawnReal(std::forward<Function>(func));
100  }
101 
102  // Submit a callable and arguments for execution. Return a future that
103  // will return the callable's result value once.
104  // The callable's arguments are copied before execution.
105  // Since the function is variadic and needs to return a result (the future),
106  // an exception is raised if the task fails spawning (which currently
107  // only occurs if the ThreadPool is shutting down).
108  template <typename Function, typename... Args,
109  typename Result = typename std::result_of<Function && (Args && ...)>::type>
110  std::future<Result> Submit(Function&& func, Args&&... args) {
111  // Trying to templatize std::packaged_task with Function doesn't seem
112  // to work, so go through std::bind to simplify the packaged signature
113  using PackagedTask = std::packaged_task<Result()>;
114  auto task = PackagedTask(std::bind(std::forward<Function>(func), args...));
115  auto fut = task.get_future();
116 
117  Status st = SpawnReal(detail::packaged_task_wrapper<Result>(std::move(task)));
118  if (!st.ok()) {
119  // This happens when Submit() is called after Shutdown()
120  throw std::runtime_error(st.ToString());
121  }
122  return fut;
123  }
