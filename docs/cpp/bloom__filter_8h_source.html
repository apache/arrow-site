115 // The BlockSplitBloomFilter is implemented using block-based Bloom filters from
116 // Putze et al.'s "Cache-,Hash- and Space-Efficient Bloom filters". The basic idea is to
117 // hash the item to a tiny Bloom filter which size fit a single cache line or smaller.
118 //
119 // This implementation sets 8 bits in each tiny Bloom filter. Each tiny Bloom
120 // filter is 32 bytes to take advantage of 32-byte SIMD instructions.

154  static uint32_t OptimalNumOfBits(uint32_t ndv, double fpp) {
155  DCHECK(fpp > 0.0 && fpp < 1.0);
156  const double m = -8.0 * ndv / log(1 - pow(fpp, 1.0 / 8));
157  uint32_t num_bits = static_cast<uint32_t>(m);
158 
159  // Handle overflow.
160  if (m < 0 || m > kMaximumBloomFilterBytes << 3) {
161  num_bits = static_cast<uint32_t>(kMaximumBloomFilterBytes << 3);
162  }
163 
164  // Round up to lower bound
165  if (num_bits < kMinimumBloomFilterBytes << 3) {
166  num_bits = kMinimumBloomFilterBytes << 3;
167  }
168 
169  // Get next power of 2 if bits is not power of 2.
170  if ((num_bits & (num_bits - 1)) != 0) {
171  num_bits = static_cast<uint32_t>(::arrow::BitUtil::NextPower2(num_bits));
172  }
173 
174  // Round down to upper bound
175  if (num_bits > kMaximumBloomFilterBytes << 3) {
176  num_bits = kMaximumBloomFilterBytes << 3;
177  }
178 
179  return num_bits;
180  }
