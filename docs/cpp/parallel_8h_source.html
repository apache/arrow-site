31 Status ParallelFor(int nthreads, int num_tasks, FUNCTION&& func) {
32  std::vector<std::thread> thread_pool;
33  thread_pool.reserve(nthreads);
34  std::atomic<int> task_counter(0);
35 
36  std::mutex error_mtx;
37  bool error_occurred = false;
38  Status error;
39 
40  for (int thread_id = 0; thread_id < nthreads; ++thread_id) {
41  thread_pool.emplace_back(
42  [&num_tasks, &task_counter, &error, &error_occurred, &error_mtx, &func]() {
43  int task_id;
44  while (!error_occurred) {
45  task_id = task_counter.fetch_add(1);
46  if (task_id >= num_tasks) {
47  break;
48  }
49  Status s = func(task_id);
50  if (!s.ok()) {
51  std::lock_guard<std::mutex> lock(error_mtx);
52  error_occurred = true;
53  error = s;
54  break;
55  }
56  }
57  });
58  }
59  for (auto&& thread : thread_pool) {
60  thread.join();
61  }
62  if (error_occurred) {
63  return error;
64  }
